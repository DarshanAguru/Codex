// Problem: https://leetcode.com/problems/container-with-most-water/
// Leetcode Problem No: 11
// Relevance: Array, Two Pointers, Greedy

import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[] arr = { 1, 5, 4, 3 };
    int res = maxArea(arr);
    System.out.println(res);
  }

  // Logic: Two Pointers (Greedy)
  // We want to maximize the area, which is `min(heights[left], heights[right]) * (right - left)`.
  // We start with pointers at the very ends of the array to maximize the width `(right - left)`.
  // To potentially find a larger area, we must increase the height. We do this by moving the
  // pointer that points to the shorter wall, as the shorter wall is the limiting factor for the area.
  //
  // Dry run for arr = [1, 5, 4, 3]:
  // 1. left=0 (h=1), right=3 (h=3). width=3. min_h=1. Area=3. max=3.
  //    Move `left` (since 1 < 3): left=1.
  // 2. left=1 (h=5), right=3 (h=3). width=2. min_h=3. Area=6. max=6.
  //    Move `right` (since 3 < 5): right=2.
  // 3. left=1 (h=5), right=2 (h=4). width=1. min_h=4. Area=4. max=6.
  //    Move `right` (since 4 < 5): right=1.
  // left == right, loop ends. Result: 6.
  static int maxArea(int[] heights) {
    int n = heights.length;
    int max = 0;

    int left = 0;
    int right = n - 1;

    // Two Pointer Approach: O(N)
    while (left < right) {
      // Calculate area based on shorter boundary
      int currentArea =
        Math.min(heights[left], heights[right]) * (right - left);
      max = Math.max(max, currentArea);

      // Move the pointer of the shorter wall to try and find a taller one
      if (heights[left] < heights[right]) {
        left++;
      } else {
        right--;
      }
    }
    return max;
  }
}
