// Problem: https://leetcode.com/problems/split-array-largest-sum/

import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[] arr = { 25, 46, 28, 49, 24 };
    int m = 4;

    int min = Arrays.stream(arr).max().getAsInt(); // Lower bound: Max pages in one book
    int max = Arrays.stream(arr).sum(); // Upper bound: Sum of all pages

    int res = binarySearch(arr, min, max, m);
    System.out.println(res);
  }

  static int countStudents(int[] arr, int pages) {
    int count = 1;
    int curr = 0;

    for (int i = 0; i < arr.length; i++) {
      if (curr + arr[i] <= pages) {
        curr += arr[i];
      } else {
        count++;
        curr = arr[i];
      }
    }
    return count;
  }

  static int binarySearch(int[] arr, int low, int high, int m) {
    if (m > arr.length) return -1;

    int ans = -1;
    while (low <= high) {
      int mid = (low + high) / 2;

      if (countStudents(arr, mid) <= m) {
        // If possible to allocate with 'mid' max pages, try smaller max
        ans = mid;
        high = mid - 1;
      } else {
        // Need more pages per student or fewer students
        low = mid + 1;
      }
    }
    return ans == -1 ? low : ans;
  }
}
