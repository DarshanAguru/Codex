# Problem: https://leetcode.com/problems/mirror-distance-of-an-integer
# Leetcode Problem No: 3783
# Relevance: Math, String

class Solution:

    # Logic: String Reversal & Absolute Difference
    # The "mirror distance" is the absolute difference between an integer and its digit-reversal.
    # 1. Convert the integer 'n' to a string to easily reverse the order of digits.
    # 2. Reverse the string using the slicing technique [::-1].
    # 3. Cast the reversed string back into an integer.
    # 4. Return the absolute difference: abs(original - mirrored).
    #
    # Example for n = 123:
    # 1. String: "123" -> Reversed: "321" -> Integer: 321.
    # 2. Difference: abs(123 - 321) = 198.
    # Result: 198
    def mirrorDistance(self, n: int) -> int:
        # Convert to string, reverse, and back to int
        mirrored = int(str(n)[::-1])
        return abs(n - mirrored)

if __name__ == "__main__":
    sol = Solution()
    
    # Example 1: n = 123
    # 123 - 321 = -198 -> abs = 198
    print(f"Example 1 Result: {sol.mirrorDistance(123)}") # Expected: 198
    
    # Example 2: n = 100
    # 100 - 001 = 99 -> abs = 99
    print(f"Example 2 Result: {sol.mirrorDistance(100)}") # Expected: 99
    
    # Example 3: n = 121
    # 121 - 121 = 0 (Palindrome)
    print(f"Example 3 Result: {sol.mirrorDistance(121)}") # Expected: 0
