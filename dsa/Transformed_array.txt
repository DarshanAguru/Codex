import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[] nums = { 3, -2, 1, 1 };
    int[] res = constructTransformedArray(nums);
    System.out.println(Arrays.toString(res));
  }

  public static int[] constructTransformedArray(int[] nums) {
    int n = nums.length;
    int[] res = new int[n];

    for (int i = 0; i < n; i++) {
      res[i] = nums[(((i + nums[i]) % n) + n) % n];
    }

    return res;
  }
}
