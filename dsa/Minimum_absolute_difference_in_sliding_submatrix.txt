// Problem: https://leetcode.com/problems/minimum-absolute-difference-in-sliding-submatrix/
// Leetcode Problem No: 3567
// Relevance: Array, Matrix, Sorting, Sliding Window

import java.util.*;

class Main {

  public static void main(String[] args) {
    int[][] grid = { { 1, 8 }, { 3, -2 } };
    int k = 2;
    int[][] res = minAbsDiff(grid, k);
    System.out.println(Arrays.deepToString(res));
  }

  // Logic: Brute Force with Sorting
  // For each possible k x k submatrix in the given m x n grid:
  // 1. Extract all k*k elements into a list.
  // 2. Sort the elements to bring close values together.
  // 3. Iterate through the sorted list and calculate the difference between
  //    adjacent distinct elements.
  // 4. The minimum of these differences is the answer for that particular submatrix.
  // 5. Special case: If all elements are the same, the result is 0.
  //
  // Dry run idea:
  // grid = [[1, 8], [3, -2]], k = 2
  // 1. Only one 2x2 submatrix: [[1, 8], [3, -2]]
  // 2. kgrid = [1, 8, 3, -2]. Sorted: [-2, 1, 3, 8]
  // 3. Differences between adjacent distinct elements:
  //    |1 - (-2)| = 3
  //    |3 - 1| = 2
  //    |8 - 3| = 5
  // 4. Minimum difference is 2.
  // 5. Result: [[2]]
  public static int[][] minAbsDiff(int[][] grid, int k) {
    int m = grid.length;
    int n = grid[0].length;
    int[][] res = new int[m - k + 1][n - k + 1];

    for (int i = 0; i + k <= m; i++) {
      for (int j = 0; j + k <= n; j++) {
        List<Integer> kgrid = new ArrayList<>();
        for (int x = i; x < i + k; x++) {
          for (int y = j; y < j + k; y++) {
            kgrid.add(grid[x][y]);
          }
        }

        int kmin = Integer.MAX_VALUE;
        Collections.sort(kgrid);
        for (int t = 1; t < kgrid.size(); t++) {
          if (kgrid.get(t).equals(kgrid.get(t - 1))) {
            continue;
          }
          kmin = Math.min(kmin, kgrid.get(t) - kgrid.get(t - 1));
        }

        if (kmin != Integer.MAX_VALUE) {
          res[i][j] = kmin;
        }
      }
    }
    return res;
  }
}
