import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    char[] tasks = { 'A', 'A', 'A', 'B', 'B', 'B' };
    int n = 2;

    int res = findMinCPUIntervals(tasks, n);
    System.out.println(res);
  }

  static int findMinCPUIntervals(char[] tasks, int n) {
    int[] taskCount = new int[26];
    int maxFreq = 0;

    for (char task : tasks) {
      int idx = task - 'A';
      taskCount[idx]++;
      maxFreq = Math.max(maxFreq, taskCount[idx]);
    }

    int maxFreqTasks = 0;
    for (int cnt : taskCount) {
      if (cnt == maxFreq) {
        maxFreqTasks++;
      }
    }

    return Math.max(tasks.length, (maxFreq - 1) * (n + 1) + maxFreqTasks);
  }
}
