// Problem: https://leetcode.com/problems/two-sum-iv-input-is-a-bst/

import java.util.*;
import java.lang.*;

class Node {
    public Node left, right;
    public int val;
    Node(int val) {
        this.val = val;
    }
}

class Main {
    public static void main(String[] args) {
        Node root = new Node(5);
        root.left = new Node(3);
        root.left.left = new Node(2);
        root.left.right = new Node(4);
        root.right = new Node(6);
        root.right.right = new Node(7);
        
        int k = 9;
        boolean res = findSum(root, k, new HashSet<>());
        System.out.println(res);
    }
    
    static boolean findSum(Node root, int target, Set<Integer> set) {
        if (root == null) return false;
        
        if (findSum(root.left, target, set)) return true;
        
        if (set.contains(target - root.val)) {
            return true;
        }
        set.add(root.val);
        
        return findSum(root.right, target, set);
    }
}
