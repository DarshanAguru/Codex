from typing import List

class Solution:

    # Logic: String Conversion (O(N * M) where N is array size, M is digits per number)
    # We want to separate all digits of each number in the array while preserving order.
    # 1. Iterate through each integer `num` in the `nums` list.
    # 2. Convert each `num` to a string to easily iterate over its individual digits.
    # 3. Convert each character back into an integer and append it to our result list.
    #
    # Dry run for nums = [13, 25, 83]:
    # 1. num = 13: str(13) -> '1', '3'. Append 1, then 3. res = [1, 3]
    # 2. num = 25: str(25) -> '2', '5'. Append 2, then 5. res = [1, 3, 2, 5]
    # 3. num = 83: str(83) -> '8', '3'. Append 8, then 3. res = [1, 3, 2, 5, 8, 3]
    # Returns [1, 3, 2, 5, 8, 3]

    def separateDigits(self, nums: List[int]) -> List[int]:
        res = []
        for num in nums:
            for digit in str(num):
                res.append(int(digit))
        return res

if __name__ == "__main__":
    sol = Solution()

    # Example 1: nums = [13, 25, 83, 77] -> Expected: [1,3,2,5,8,3,7,7]
    nums1 = [13, 25, 83, 77]
    print(f"Result for {nums1}: {sol.separateDigits(nums1)}")

    # Example 2: nums = [7, 1, 3, 9] -> Expected: [7,1,3,9]
    nums2 = [7, 1, 3, 9]
    print(f"Result for {nums2}: {sol.separateDigits(nums2)}")
