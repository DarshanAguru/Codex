// Problem: https://leetcode.com/problems/diameter-of-binary-tree/

import java.util.*;
import java.lang.*;

class Node {
    public Node left, right;
    public int val;
    Node(int val) {
        this.val = val;
    }
}

class Main {
    static int dia;
    
    public static void main(String[] args) {
        Node root = new Node(1);
        root.left = new Node(2);
        root.right = new Node(3);
        root.left.left = new Node(4);
        root.left.right = new Node(5);
        
        dia = 0;
        findDiameterOfTree(root);
        System.out.println("Diameter: " + dia);
    }
    
    static int findDiameterOfTree(Node root) {
        if (root == null) return 0;
        
        int leftH = findDiameterOfTree(root.left);
        int rightH = findDiameterOfTree(root.right);
        
        dia = Math.max(dia, leftH + rightH);
        
        return 1 + Math.max(leftH, rightH);
    }
}
