// Problem: https://leetcode.com/problems/diameter-of-binary-tree/
// Leetcode Problem No: 543
// Relevance: Tree, Depth-First Search, Recursion

import java.lang.*;
import java.util.*;

class Node {

  public Node left, right;
  public int val;

  Node(int val) {
    this.val = val;
  }
}

class Main {

  static int dia;

  public static void main(String[] args) {
    Node root = new Node(1);
    root.left = new Node(2);
    root.right = new Node(3);
    root.left.left = new Node(4);
    root.left.right = new Node(5);

    dia = 0;
    findDiameterOfTree(root);
    System.out.println("Diameter: " + dia);
  }

  // Logic: DFS / Post-order Traversal
  // The diameter of a binary tree is the length of the longest path between any two nodes.
  // This path may or may not pass through the root. It's simply the maximum value of
  // (Left Subtree Height + Right Subtree Height) among all nodes in the tree.
  // We compute the height of the left and right subtrees recursively. At each node,
  // we update our global maximum diameter `dia = Math.max(dia, leftH + rightH)`.
  // The function itself returns the height of the current subtree `1 + Math.max(leftH, rightH)`.
  //
  // Dry run for root = 1, L=2 (L=4, R=5), R=3:
  // find(4) -> left=0, right=0. dia=max(0, 0)=0. returns 1.
  // find(5) -> left=0, right=0. dia=max(0, 0)=0. returns 1.
  // find(2) -> left=1, right=1. dia=max(0, 2)=2. returns 1+max(1,1)=2.
  // find(3) -> left=0, right=0. dia=max(2, 0)=2. returns 1.
  // find(1) -> left=2 (from 2), right=1 (from 3). dia=max(2, 2+1=3)=3. returns 1+max(2,1)=3.
  // Final diameter is 3. (Path: 4->2->1->3).
  static int findDiameterOfTree(Node root) {
    if (root == null) return 0;

    int leftH = findDiameterOfTree(root.left);
    int rightH = findDiameterOfTree(root.right);

    dia = Math.max(dia, leftH + rightH);

    return 1 + Math.max(leftH, rightH);
  }
}
