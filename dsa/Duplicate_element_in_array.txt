// Leetcode Problem No: 287
// Relevance: Hash Table

import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[] arr = { 1, 2, 3, 4 };
    // Note: This solution uses O(N) space.
    // Floyd's Cycle Finding (Tortoise and Hare) can solve this in O(1) space.
    String duplicate = null;

    HashSet<Integer> hs = new HashSet<>();

    for (int i : arr) {
      if (hs.contains(i)) {
        duplicate = String.valueOf(i);
      }
      hs.add(i);
    }

    System.out.println("Duplicate Element " + duplicate);
  }
}
