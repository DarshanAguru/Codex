# Problem: https://leetcode.com/problems/robot-return-to-origin/
# Leetcode Problem No: 657
# Relevance: String, Simulation

class Solution:

    # Logic: 2D Coordinate Simulation
    # 1. Starting at (0,0), represent the robot's position as [x, y].
    # 2. Iterate through each move in the string:
    #    - 'U' (Up): Increment Y (Vertical)
    #    - 'D' (Down): Decrement Y (Vertical)
    #    - 'L' (Left): Decrement X (Horizontal)
    #    - 'R' (Right): Increment X (Horizontal)
    # 3. After all moves, check if the position is back at [0,0].
    #
    # Dry run for moves = "UD":
    # 1. Init: pos = [0,0]
    # 2. move = 'U': pos = [0,1]
    # 3. move = 'D': pos = [0,0]
    # 4. Final: pos == [0,0] -> True
    
    def judgeCircle(self, moves: str) -> bool:
        # pos[0] is vertical (U/D), pos[1] is horizontal (L/R)
        # Note: Usually pos[0] is x and pos[1] is y, but let's stay consistent with moves.
        pos = [0, 0]
        for move in moves:
            if move == "U":
                pos[0] += 1
            elif move == "D":
                pos[0] -= 1
            elif move == "L":
                pos[1] -= 1
            elif move == "R":
                pos[1] += 1
        
        return pos == [0, 0]

if __name__ == "__main__":
    sol = Solution()
    
    # Example 1: moves = "UD"
    print(f"Example 1 (UD): {sol.judgeCircle('UD')}") # Expected: True
    
    # Example 2: moves = "LL"
    print(f"Example 2 (LL): {sol.judgeCircle('LL')}") # Expected: False
