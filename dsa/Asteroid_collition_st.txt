import java.lang.*;
import java.util.*;
import java.util.stream.*;

class Main {

  public static void main(String[] args) {
    int[] asteroids = { 3, 2, -5 };

    int[] res = getStateAfterCollision(asteroids);
    System.out.println(Arrays.toString(res));
  }

  static int[] getStateAfterCollision(int[] asteroids) {
    Stack<Integer> st = new Stack<>();

    for (int ast : asteroids) {
      if (ast > 0) {
        // Positive asteroid always moves right, just push
        st.push(ast);
      } else {
        // Negative asteroid moving left. Check for collisions with top of stack (right-moving asteroids).
        // While stack has a positive asteroid (st.peek() > 0) and it is smaller than current (|ast| > st.peek())
        while (!st.isEmpty() && st.peek() > 0 && st.peek() < -ast) {
          st.pop(); // Positive asteroid explodes
        }

        if (!st.isEmpty() && st.peek() == -ast) {
          // Same size: both explode
          st.pop();
        } else if (st.isEmpty() || st.peek() < 0) {
          // Stack empty or only negative asteroids left: current one survives
          st.push(ast);
        }
      }
    }

    return st
      .stream()
      .mapToInt(i -> i)
      .toArray();
  }
}
