from typing import List
from collections import defaultdict


class UnionFind:
    def __init__(self, n):
        self.fa = list(range(n))
        self.rank = [0] * n

    def find(self, x):
        if self.fa[x] != x:
            self.fa[x] = self.find(self.fa[x])
        return self.fa[x]

    def union(self, x, y):
        x = self.find(x)
        y = self.find(y)
        if x == y:
            return
        if self.rank[x] < self.rank[y]:
            x, y = y, x
        self.fa[y] = x
        if self.rank[x] == self.rank[y]:
            self.rank[x] += 1


class Solution:

    # Logic: Union-Find (Disjoint Set Union)
    # Indices connected via allowedSwaps can freely share elements among themselves.
    # So, within each connected component, we can arrange source values in any order.
    #
    # Step 1: Build a Union-Find over indices, unioning all allowedSwap pairs.
    # Step 2: Group source values by their component root using a nested defaultdict:
    #         sets[root][value] = count of that value available in this component.
    # Step 3: For each index i (in order), check if target[i] is available in the
    #         same component as i. If yes, consume one occurrence (decrement count).
    #         If not, it contributes 1 to the Hamming distance.
    # Return the total Hamming distance.

    def minimumHammingDistance(
        self,
        source: List[int],
        target: List[int],
        allowedSwaps: List[List[int]],
    ) -> int:
        n = len(source)
        uf = UnionFind(n)
        for a, b in allowedSwaps:
            uf.union(a, b)

        sets = defaultdict(lambda: defaultdict(int))
        for i in range(n):
            f = uf.find(i)
            sets[f][source[i]] += 1

        ans = 0
        for i in range(n):
            f = uf.find(i)
            if sets[f][target[i]] > 0:
                sets[f][target[i]] -= 1
            else:
                ans += 1
        return ans


if __name__ == "__main__":
    sol = Solution()

    # Example 1: source=[1,2,3,4], target=[2,1,4,5], allowedSwaps=[[0,1],[2,3]] -> Expected: 1
    source = [1, 2, 3, 4]
    target = [2, 1, 4, 5]
    allowedSwaps = [[0, 1], [2, 3]]
    print(f"Result: {sol.minimumHammingDistance(source, target, allowedSwaps)}")

    # Example 2: source=[1,2,3,4], target=[1,3,2,4], allowedSwaps=[] -> Expected: 2
    source = [1, 2, 3, 4]
    target = [1, 3, 2, 4]
    allowedSwaps = []
    print(f"Result: {sol.minimumHammingDistance(source, target, allowedSwaps)}")

    # Example 3: source=[5,1,2,4,3], target=[1,5,4,2,3], allowedSwaps=[[0,4],[4,2],[1,3],[1,4]] -> Expected: 0
    source = [5, 1, 2, 4, 3]
    target = [1, 5, 4, 2, 3]
    allowedSwaps = [[0, 4], [4, 2], [1, 3], [1, 4]]
    print(f"Result: {sol.minimumHammingDistance(source, target, allowedSwaps)}")
