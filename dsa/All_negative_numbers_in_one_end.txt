// Leetcode Problem No: 905 (Similar Concept: Sort Array By Parity)
// Relevance: Array, Two Pointers, Partition

import java.util.ArrayList;

class Main {

  // Logic: Array Partitioning using Extra Space
  // The goal is to separate positive and negative numbers. This solution uses O(N) space
  // via a `temp` array.
  // First, we iterate over the elements and add all positive numbers (>= 0) into `temp`.
  // Then, we iterate again and add all negative numbers (< 0).
  // Finally, we copy the updated `temp` array back to the original `arr`.
  // (Note: To do this in O(1) space, a two-pointer approach akin to QuickSort partitioning
  // or cycle sort can be used, though preserving order in O(1) space is generally O(N^2)).
  //
  // Dry run for arr = [1, -1, 3, -7]:
  // 1. Positives pass: temp[0]=1, temp[1]=3. (l=2)
  // 2. Negatives pass: temp[2]=-1, temp[3]=-7. (l=4)
  // temp becomes [1, 3, -1, -7]. We copy this back to arr.
  public static void main(String[] args) {
    int[] arr = { 1, -1, 3, 2, -7, -5, 11, 6 };
    // Using an auxiliary array method (O(N) space)
    // To do this in O(1) space, one would use a Two-Pointer partition (similar to QuickSort)
    int[] temp = new int[arr.length];

    int l = 0;

    // First Pass: Collect all positive numbers (including 0)
    for (int i = 0; i < arr.length; i++) {
      if (arr[i] >= 0) {
        temp[l++] = arr[i];
      }
    }

    // Second Pass: Collect all negative numbers
    // Note: This logic puts Positives FIRST.
    // The problem title says "All negative numbers in one end", traditionally at start,
    // but this code puts positives at start. The logic implies separation.
    for (int i = 0; i < arr.length; i++) {
      if (arr[i] < 0) {
        temp[l++] = arr[i];
      }
    }

    // Copy back to original array
    System.arraycopy(temp, 0, arr, 0, arr.length);

    // Print Result
    for (int n : arr) {
      System.out.print(n + " ");
    }
  }
}
