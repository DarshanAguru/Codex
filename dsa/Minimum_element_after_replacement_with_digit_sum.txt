# Problem: https://leetcode.com/problems/minimum-element-after-replacement-with-digit-sum/
# Leetcode Problem No: 3300
# Relevance: Array, Math

from typing import List

class Solution:

    # Logic: String-based Digit Sum and Min Finding (O(N * log10(M)) Time, O(log10(M)) Space)
    # where N = number of elements in nums, and M = maximum value in nums.
    #
    # 1. We want to replace each element in the array with its digit sum, then return the minimum element of the resulting array.
    # 2. To get the digit sum of `n`, we convert it to a string `str(n)`, map each character to an integer `int`, and sum them.
    # 3. We maintain a running minimum `res` initialized to positive infinity (`float('inf')`).
    # 4. We iterate through each `num` in `nums`, update `res = min(res, getDigitSum(num))`, and return the final minimum `res`.
    #
    # Dry run for nums = [10, 12, 13, 14]:
    # 1. Initialize res = inf.
    # 2. Process num = 10:
    #    - getDigitSum(10) -> sum(map(int, "10")) -> sum([1, 0]) -> 1.
    #    - res = min(inf, 1) = 1.
    # 3. Process num = 12:
    #    - getDigitSum(12) -> sum(map(int, "12")) -> sum([1, 2]) -> 3.
    #    - res = min(1, 3) = 1.
    # 4. Process num = 13:
    #    - getDigitSum(13) -> sum(map(int, "13")) -> sum([1, 3]) -> 4.
    #    - res = min(1, 4) = 1.
    # 5. Process num = 14:
    #    - getDigitSum(14) -> sum(map(int, "14")) -> sum([1, 4]) -> 5.
    #    - res = min(1, 5) = 1.
    # 6. Returns 1.

    def minElement(self, nums: List[int]) -> int:
        
        def getDigitSum(n: int) -> int:
            return sum(map(int,(str(n))))
        
        res = float('inf')

        for num in nums:
            res = min(res, getDigitSum(num))
        
        return res

if __name__ == "__main__":
    sol = Solution()

    # Example 1: nums = [10, 12, 13, 14] -> Expected: 1
    nums1 = [10, 12, 13, 14]
    print(f"Result for {nums1}: {sol.minElement(nums1)}")

    # Example 2: nums = [999, 19, 199] -> Expected: 10
    nums2 = [999, 19, 199]
    print(f"Result for {nums2}: {sol.minElement(nums2)}")
