import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int n = 5;
    boolean res = hasAlternatingBits(n);
    System.out.println(res);
  }

  static boolean hasAlternatingBits(int n) {
    int prev = n & 1;
    n >>= 1;

    while (n != 0) {
      int curr = n & 1;
      if (prev == curr) {
        return false;
      }
      prev = curr;
      n >>= 1;
    }
    return true;
  }
}
