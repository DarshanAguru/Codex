import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int n = 6;
    int set = 0;

    // Count set bits using Brian Kernighanâ€™s Algorithm
    // Iterates only as many times as there are set bits
    while (n != 0) {
      n = n & (n - 1);
      set++;
    }

    System.out.println(set);
  }
}
