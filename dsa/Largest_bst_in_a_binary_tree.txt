// Problem: https://leetcode.com/problems/largest-bst-subtree/
// Leetcode Problem No: 333
// Relevance: Tree, Depth-First Search, Binary Search Tree, Recursion

import java.lang.*;
import java.util.*;

class Node {

  public Node left, right;
  public int val;

  Node(int val) {
    this.val = val;
  }
}

class BSTInfo {

  public int min;
  public int max;
  public int size;

  BSTInfo(int min, int max, int size) {
    this.min = min;
    this.max = max;
    this.size = size;
  }
}

class Main {

  public static void main(String[] args) {
    Node root = new Node(5);
    root.left = new Node(2);
    root.left.left = new Node(1);
    root.left.right = new Node(3);
    root.right = new Node(4);

    int res = findLargestBSTInTree(root);
    System.out.println("Largest BST Size: " + res);
  }

  // Logic: Post-order Traversal (Bottom-Up)
  // To find the largest BST subtree, we traverse the tree from the bottom up.
  // For each node, we need to know 3 things from its left and right children:
  // 1. Minimum value in the subtree.
  // 2. Maximum value in the subtree.
  // 3. Size of the largest BST found so far in that subtree.
  // A subtree rooted at `node` is a valid BST if:
  //   `left_subtree.max < node.val` AND `right_subtree.min > node.val`.
  // If it is a BST, we return a new info object with updated min, max, and `size = left.size + right.size + 1`.
  // If it is NOT a BST, we return `min = -Infinity`, `max = Infinity` (ensuring parent also fails BST check),
  // and `size = max(left.size, right.size)`.
  //
  // Dry run for tree = 5, L=2(L=1, R=3), R=4:
  // Node(1) (leaf): It's a BST. min=1, max=1, size=1.
  // Node(3) (leaf): It's a BST. min=3, max=3, size=1.
  // Node(2): Left max(1) < 2 AND Right min(3) > 2. Yes, it's a BST. min=1, max=3, size=1+1+1=3.
  // Node(4) (leaf): It's a BST. min=4, max=4, size=1.
  // Node(5) (root): Left max(3) < 5... BUT Right min(4) is NOT > 5 (fail!).
  //                 Not a BST. Return size = max(Left.size(3), Right.size(1)) = 3.
  // Largest BST size is 3 (rooted at Node(2)).
  static int findLargestBSTInTree(Node root) {
    return helper(root).size;
  }

  static BSTInfo helper(Node root) {
    if (root == null) {
      return new BSTInfo(Integer.MAX_VALUE, Integer.MIN_VALUE, 0);
    }

    BSTInfo left = helper(root.left);
    BSTInfo right = helper(root.right);

    // Check if current node forms a BST with left and right subtrees
    if (left.max < root.val && right.min > root.val) {
      return new BSTInfo(
        Math.min(left.min, root.val),
        Math.max(right.max, root.val),
        1 + left.size + right.size
      );
    }

    // Not a BST, effectively discard by returning range [-Inf, Inf] but larger size
    return new BSTInfo(
      Integer.MIN_VALUE,
      Integer.MAX_VALUE,
      Math.max(left.size, right.size)
    );
  }
}
