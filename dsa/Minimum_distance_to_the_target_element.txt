from typing import List

class Solution:

    # Logic: Radial Expansion
    # 1. We want to find the minimum distance from 'start', so we search outwards radially.
    # 2. We expand a distance 'd' starting from 0 up to 'n'.
    # 3. For each distance 'd', we check both 'start - d' and 'start + d' simultaneously.
    # 4. This guarantees that the first time we encounter the 'target', it must be at the absolute minimum possible distance.
    #
    # Example for nums = [1, 2, 3, 4, 5], target = 5, start = 3:
    # 1. d=0: left=3 (val=4), right=3 (val=4)
    # 2. d=1: left=2 (val=3), right=4 (val=5). right matches target!
    # Result: 1
    def getMinDistance(self, nums: List[int], target: int, start: int) -> int:
        n = len(nums)

        for d in range(n):
            left = start - d
            right = start + d

            if left >= 0 and nums[left] == target:
                return d

            if right < n and nums[right] == target:
                return d
        
        return n
