# Problem: https://leetcode.com/problems/find-minimum-in-rotated-sorted-array/
# Leetcode Problem No: 153
# Relevance: Array, Binary Search

from typing import List

class Solution:

    # Logic: Binary Search (O(log N) Time, O(1) Space)
    # 1. Initialize `low = 0` and `high = n - 1`.
    # 2. While `low < high`:
    #    - Calculate `mid = (low + high) // 2`.
    #    - If `nums[mid] < nums[high]`:
    #        - The right part is sorted, and the minimum must be in the left part (including `mid`).
    #        - Set `high = mid`.
    #    - Else (`nums[mid] >= nums[high]`):
    #        - The left part is sorted, and the minimum must be in the right part (excluding `mid`).
    #        - Set `low = mid + 1`.
    # 3. After the loop, `low == high` will point to the minimum element.
    #
    # Dry run for nums = [3, 4, 5, 1, 2]:
    # 1. low = 0, high = 4, mid = 2 (nums[2] = 5):
    #    - 5 > 2, so low = 3.
    # 2. low = 3, high = 4, mid = 3 (nums[3] = 1):
    #    - 1 < 2, so high = 3.
    # 3. low = 3, high = 3. Loop ends.
    # Returns nums[3] = 1.
    
    def findMin(self, nums: List[int]) -> int:
        n = len(nums)
        low, high = 0, n-1
        while low < high:
            mid = (low + high) // 2
            if nums[mid] < nums[high]:
                high = mid
            else:
                low = mid+1
        
        return nums[low]

if __name__ == "__main__":
    sol = Solution()

    # Example 1: nums = [3, 4, 5, 1, 2] -> Expected: 1
    nums1 = [3, 4, 5, 1, 2]
    print(f"Result for {nums1}: {sol.findMin(nums1)}")

    # Example 2: nums = [4, 5, 6, 7, 0, 1, 2] -> Expected: 0
    nums2 = [4, 5, 6, 7, 0, 1, 2]
    print(f"Result for {nums2}: {sol.findMin(nums2)}")

    # Example 3: nums = [11, 13, 15, 17] -> Expected: 11
    nums3 = [11, 13, 15, 17]
    print(f"Result for {nums3}: {sol.findMin(nums3)}")
