// Problem: https://leetcode.com/problems/happy-number/
// Leetcode Problem No: 202
// Relevance: Hash Table, Math, Two Pointers

import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int n = 7;
    boolean res = isHappy(n);
    System.out.println(res);
  }

  // Logic: Floyd's Cycle Detection Algorithm (Fast & Slow Pointers)
  // Finding if a number is "Happy" simulates moving through a sequence of numbers.
  // The sequence either eventually reaches 1, or gets stuck in a cycle (an infinite loop).
  // We can use a `slow` pointer (moves 1 step: gets next sum of squares) and
  // a `fast` pointer (moves 2 steps).
  // If `fast` reaches 1, the number is happy. If `slow == fast` before reaching 1,
  // there is a cycle, so it's not a happy number.
  //
  // Dry run for n = 7 (Expected: True):
  // getNext(7) = 49. getNext(49) = 16+81=97. getNext(97) = 81+49=130.
  // getNext(130) = 1+9+0=10. getNext(10) = 1+0=1.
  // Instead of storing all seen numbers in a HashSet, the tortoise & hare pointers
  // will eventually land on `fast == 1` correctly in O(1) space.
  static boolean isHappy(int n) {
    int slow = n;
    int fast = getNext(n);

    while (fast != 1 && slow != fast) {
      slow = getNext(slow);
      fast = getNext(getNext(fast));
    }
    return fast == 1;
  }

  static int getNext(int n) {
    int sum = 0;
    while (n > 0) {
      int digit = n % 10;
      sum += digit * digit;
      n /= 10;
    }
    return sum;
  }
}
