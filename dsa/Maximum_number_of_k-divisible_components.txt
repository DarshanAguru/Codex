// Problem: https://leetcode.com/problems/maximum-number-of-k-divisible-components/

import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int n = 7;
    int[][] edges = {
      { 0, 1 },
      { 0, 2 },
      { 1, 3 },
      { 1, 4 },
      { 2, 5 },
      { 2, 6 },
    };
    int[] values = { 3, 0, 6, 1, 5, 2, 1 };
    int k = 3;

    int res = maxKDivisibleComponents(n, edges, values, k);
    System.out.println("Result: " + res);
  }

  static int maxKDivisibleComponents(
    int n,
    int[][] edges,
    int[] values,
    int k
  ) {
    if (n < 2) return 1;

    List<List<Integer>> adj = new ArrayList<>();
    for (int i = 0; i < n; i++) adj.add(new ArrayList<>());

    int[] degree = new int[n];
    for (int[] e : edges) {
      adj.get(e[0]).add(e[1]);
      adj.get(e[1]).add(e[0]);
      degree[e[0]]++;
      degree[e[1]]++;
    }

    // Convert to long to prevent overflow
    long[] longValues = new long[n];
    for (int i = 0; i < n; i++) longValues[i] = values[i];

    Deque<Integer> queue = new LinkedList<>();
    for (int i = 0; i < n; i++) {
      if (degree[i] == 1) queue.offer(i);
    }

    int components = 0;
    while (!queue.isEmpty()) {
      int curr = queue.poll();
      degree[curr]--;

      long carry = 0;
      if (longValues[curr] % k == 0) {
        components++;
      } else {
        carry = longValues[curr];
      }

      for (int neighbor : adj.get(curr)) {
        if (degree[neighbor] > 0) {
          degree[neighbor]--;
          longValues[neighbor] += carry;
          if (degree[neighbor] == 1) {
            queue.offer(neighbor);
          }
        }
      }
    }
    return components;
  }
}
