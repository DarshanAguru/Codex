import math
import heapq
from typing import List

class SparseTable:
    def __init__(self, arr, func = min):
        self.n = len(arr)
        self.P = int(math.log2(self.n)) + 1
        self.f = func
        self.st = [[0] * self.P for _ in range(self.n)]

        for i in range(self.n):
            self.st[i][0] = arr[i]
        
        j = 1
        while (1 << j) <= self.n:
            i = 0
            while (i + (1 << j) - 1) < self.n:
                self.st[i][j] = self.f(self.st[i][j-1], self.st[i + (1 << (j-1))][j-1])
                i += 1
            j += 1
    
    def query(self, l, r):
        size = r - l + 1
        k = int(math.log2(size))
        return self.f(self.st[l][k] , self.st[r - (1 << k) + 1][k])

class Solution:

    # Logic: Greedy Heap Search with Sparse Table (O((N + K) log N) Time, O(N log N) Space)
    # 1. Unlike Part I, we must select `k` distinct subarrays.
    # 2. The maximum possible subarray value is obtained from the whole array `[0, n - 1]`.
    # 3. We can explore smaller subarrays (candidates) by starting from the full array and
    #    greedily shrinking the boundary by 1 from either the left or the right.
    # 4. We use a max-heap (implemented using Python's `heapq` with negated values) to store potential subarrays.
    # 5. To answer Range Maximum/Minimum Queries in O(1) time, we construct two Sparse Tables:
    #    - `mxst` for range maximums.
    #    - `mnst` for range minimums.
    # 6. We start by pushing the full interval `(0, n - 1)` into the heap.
    # 7. For `k` iterations:
    #    - Pop the subarray `(l, r)` with the largest value.
    #    - Add its value to our total result.
    #    - Push the left-shrunk candidate `(l + 1, r)` and right-shrunk candidate `(l, r - 1)` into the heap,
    #      using a `seen` set to avoid duplicate states.
    #
    # Dry run for nums = [1, 5, 2], k = 2:
    # 1. mxst = SparseTable(nums, max), mnst = SparseTable(nums, min)
    #    - query(0, 2): max=5, min=1 -> value = 4.
    #    - query(1, 2): max=5, min=2 -> value = 3.
    #    - query(0, 1): max=5, min=1 -> value = 4.
    # 2. pq = [(-4, 0, 2)], seen = {(0, 2)}, res = 0.
    # 3. Iteration 1:
    #    - Pop (-4, 0, 2). res += 4 -> res = 4.
    #    - Candidate 1: (1, 2). Val = 3. Push (-3, 1, 2) to pq, add to seen.
    #    - Candidate 2: (0, 1). Val = 4. Push (-4, 0, 1) to pq, add to seen.
    # 4. Iteration 2:
    #    - Pop (-4, 0, 1). res += 4 -> res = 8.
    #    - Push candidates (1, 1) and (0, 0) (both have val=0) to pq.
    # 5. Return res = 8.

    def maxTotalValue(self, nums: List[int], k: int) -> int:
        n = len(nums)
        mxst = SparseTable(nums, max)
        mnst = SparseTable(nums)
        res = 0
        pq = []
        l, r = 0, n - 1
        seen = set()
        heapq.heappush(pq, (-(mxst.query(l, r) - mnst.query(l, r)), l, r))
        seen.add((l, r))
        
        for _ in range(k):
            v, l, r = heapq.heappop(pq)
            res += -v

            if l + 1 <= r and (l + 1, r) not in seen:
                seen.add((l + 1, r))
                heapq.heappush(pq, (-(mxst.query(l + 1, r) - mnst.query(l + 1, r)), l + 1, r))
            
            if r - 1 >= l and (l, r - 1) not in seen:
                seen.add((l, r - 1))
                heapq.heappush(pq, (-(mxst.query(l, r - 1) - mnst.query(l, r - 1)), l, r - 1))

        return res

if __name__ == "__main__":
    sol = Solution()

    # Example 1: nums = [1, 5, 2], k = 2 -> Expected: 8
    nums1 = [1, 5, 2]
    k1 = 2
    print(f"Result: {sol.maxTotalValue(nums1, k1)}")

    # Example 2: nums = [2, 3, 1, 5, 4], k = 4 -> Expected: 15
    nums2 = [2, 3, 1, 5, 4]
    k2 = 4
    print(f"Result: {sol.maxTotalValue(nums2, k2)}")
