// Leetcode Problem No: 448
// Relevance: Array, HashTable

import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[] nums = { 4, 3, 2, 7, 8, 2, 3, 1 };

    List<Integer> res = findDisappearedNumbers(nums);

    System.out.println(res);
  }

  // Logic: Cyclic Sort
  // We have numbers from 1 to `N` in an array of size `N`.
  // We can sort them in O(N) time and O(1) space by swapping each element to its correct index
  // (e.g., number 4 belongs at index 3).
  // Once the cycle sort is done, we iterate through the array. If `arr[k] != k + 1`,
  // then the number `k + 1` is missing (disappeared).
  //
  // Dry run for nums = [4, 3, 2, 7, 8, 2, 3, 1]:
  // i=0: nums[0]=4. Correct index is 3. Swap nums[0] and nums[3]. -> [7, 3, 2, 4, 8, 2, 3, 1]
  // i=0: nums[0]=7. Swap with nums[6]. -> [3, 3, 2, 4, 8, 2, 7, 1]
  // i=0: nums[0]=3. Swap with nums[2]. -> [2, 3, 3, 4, 8, 2, 7, 1]
  // i=0: nums[0]=2. Swap with nums[1]. -> [3, 2, 3, 4, 8, 2, 7, 1]
  // i=0: nums[0]=3. Correct index 2 is already occupied by 3 (nums[2]==3). They are same, move on (i++).
  // ... after full sorting ...
  // nums becomes [1, 2, 3, 4, 3, 2, 7, 8].
  // Finding missing: index 4 has 3 (expected 5). Add 5. Index 5 has 2 (expected 6). Add 6.
  // Result is [5, 6].
  public static List<Integer> findDisappearedNumbers(int[] nums) {
    List<Integer> res = new ArrayList<>();
    int i = 0;
    while (i < nums.length) {
      int j = nums[i] - 1;
      if (nums[i] != nums[j]) {
        int tmp = nums[i];
        nums[i] = nums[j];
        nums[j] = tmp;
      } else {
        i++;
      }
    }

    for (int k = 0; k < nums.length; k++) {
      if (nums[k] != k + 1) {
        res.add(k + 1);
      }
    }
    return res;
  }
}
