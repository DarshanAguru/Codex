// Problem: https://leetcode.com/problems/two-sum/
// Leetcode Problem No: 1
// Relevance: Array, Two Pointers, Hash Table

import java.lang.*;
import java.util.*;

class Main {

  // Logic: Two Pointers (Requires Sorting)
  // We want to find if there exist two elements in the array that sum to `target`.
  // Note: Finding EXACT indices usually requires a HashMap (O(N) time and space).
  // This approach SORTs the array first, making it O(N log N) time but O(1) space.
  // 1. Sort the array.
  // 2. Use two pointers: `left` at 0, `right` at `n-1`.
  // 3. Compute `sum = arr[left] + arr[right]`.
  //    - If `sum == target`, we found a pair.
  //    - If `sum < target`, we need a larger sum, so we shift `left++`.
  //    - If `sum > target`, we need a smaller sum, so we shift `right--`.
  //
  // Dry run for arr = [2, 6, 5, 8, 11], target = 15:
  // Sorted: [2, 5, 6, 8, 11]
  // l=0(2), r=4(11). sum=13. 13 < 15. l=1(5).
  // l=1(5), r=4(11). sum=16. 16 > 15. r=3(8).
  // l=1(5), r=3(8). sum=13. 13 < 15. l=2(6).
  // l=2(6), r=3(8). sum=14. 14 < 15. l=3(8).
  // Loop ends (l < r is false). target not found here? Wait, 15 isn't in arr!
  // Output: NO.
  public static void main(String[] args) {
    int[] arr = { 2, 6, 5, 8, 11 };
    int target = 15;

    boolean found = false;

    // Sort is required for Two Pointer approach
    Arrays.sort(arr);

    int left = 0;
    int right = arr.length - 1;

    while (left < right) {
      int sum = arr[left] + arr[right];

      if (sum == target) {
        found = true;
        break;
      } else if (sum < target) {
        left++;
      } else {
        right--;
      }
    }

    if (found) {
      System.out.println("YES");
    } else {
      System.out.println("NO");
    }
  }
}
