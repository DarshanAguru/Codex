// Problem: https://leetcode.com/problems/middle-of-the-linked-list/

import java.lang.*;
import java.util.*;

class Node {

  int val;
  Node next;

  Node(int val) {
    this.val = val;
    this.next = null;
  }
}

class LL {

  public Node head;

  public void add(int n) {
    if (head == null) {
      head = new Node(n);
      return;
    }
    Node temp = head;
    while (temp.next != null) {
      temp = temp.next;
    }
    temp.next = new Node(n);
  }

  public void printList() {
    Node temp = head;
    System.out.print("[ ");
    while (temp != null) {
      System.out.print(temp.val + " ");
      temp = temp.next;
    }
    System.out.println("]");
  }
}

class Main {

  public static void main(String[] args) {
    LL list = new LL();
    list.add(1);
    list.add(2);
    list.add(3);
    list.add(4);
    list.add(5);

    list.printList();

    int res = findMiddle(list.head);
    System.out.println("Middle Element: " + res);
  }

  // Tortoise and Hare approach
  static int findMiddle(Node head) {
    Node slow = head;
    Node fast = head;

    while (fast != null && fast.next != null) {
      slow = slow.next;
      fast = fast.next.next;
    }
    return slow.val;
  }
}
