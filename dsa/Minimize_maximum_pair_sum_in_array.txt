// Date: 24/01/2026
// Problem: https://leetcode.com/problems/minimize-maximum-pair-sum-in-array/
// Leetcode Problem No: 1877
// Relevance: Array, Sorting, Two Pointers, Greedy

import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[] nums = { 3, 5, 4, 2, 4, 6 };
    int res = minPairSum(nums);
    System.out.println(res);
  }

  // Logic: Sorting + Greedy Two Pointers
  // We want to pair up numbers such that the maximum pair sum is minimized.
  // To minimize the maximum pair sum, we must pair the largest remaining number
  // with the smallest remaining number.
  // If we pair a large number with another large number, their sum will be huge.
  // 1. Sort the entire array.
  // 2. Use a `left` pointer at the start (smallest) and a `right` pointer
  //    at the end (largest).
  // 3. Pair `nums[left]` with `nums[right]`, then move the pointers inwards.
  // 4. Update the maximum sum seen so far across all pairs.
  //
  // Dry run for nums = [3, 5, 4, 2, 4, 6]:
  // Sort: [2, 3, 4, 4, 5, 6].
  // 1. left=0 (2), right=5 (6). sum = 2+6 = 8. max=8.
  // 2. left=1 (3), right=4 (5). sum = 3+5 = 8. max=8.
  // 3. left=2 (4), right=3 (4). sum = 4+4 = 8. max=8.
  // Result is 8.
  public static int minPairSum(int[] nums) {
    int n = nums.length;
    Arrays.sort(nums);

    int res = 0,
      left = 0,
      sum = 0;
    int right = n;

    while (left < right) {
      sum = nums[left++] + nums[--right];
      res = Math.max(res, sum);
    }

    return res;
  }
}
