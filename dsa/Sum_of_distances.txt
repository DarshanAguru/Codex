# Problem: https://leetcode.com/problems/sum-of-distances/
# Leetcode Problem No: 2615
# Relevance: Array, Hash Map, Prefix Sum

from typing import List
from collections import defaultdict


class Solution:

    # Logic: Prefix Sum per value group
    # For each unique value in nums, collect all indices where it appears.
    # The answer for index pos[i] is the sum of absolute differences from pos[i]
    # to every other index in the same group.
    #
    # For a sorted list of positions [p0, p1, ..., pm-1], at position pos[i]:
    #   - left contribution  = pos[i] * i - left_sum
    #     (i elements to the left, each contributing pos[i] - pos[j])
    #   - right contribution = right_sum - pos[i] * (m - i - 1)
    #     (m-i-1 elements to the right, each contributing pos[j] - pos[i])
    # where left_sum = sum of pos[0..i-1] and right_sum = total - left_sum - pos[i].
    # Since mp[v] is built via enumeration (left to right), it is already sorted.

    def distance(self, nums: List[int]) -> List[int]:
        n = len(nums)
        ans = [0] * n

        mp = defaultdict(list)
        for i, v in enumerate(nums):
            mp[v].append(i)

        for pos in mp.values():
            total = sum(pos)
            left_sum = 0
            m = len(pos)

            for i in range(m):
                right_sum = total - left_sum - pos[i]

                left = pos[i] * i - left_sum
                right = right_sum - pos[i] * (m - i - 1)

                ans[pos[i]] = left + right

                left_sum += pos[i]

        return ans


if __name__ == "__main__":
    sol = Solution()

    # Example 1: nums = [1,3,1,1,2] -> Expected Result: [5,0,3,4,0]
    nums = [1, 3, 1, 1, 2]
    print(f"Result: {sol.distance(nums)}")

    # Example 2: nums = [0,5,3] -> Expected Result: [0,0,0]
    nums = [0, 5, 3]
    print(f"Result: {sol.distance(nums)}")

    # Example 3: nums = [1,1,1] -> Expected Result: [3,2,3]
    nums = [1, 1, 1]
    print(f"Result: {sol.distance(nums)}")
