# Problem: https://leetcode.com/problems/find-the-prefix-common-array-of-two-arrays/
# Leetcode Problem No: 2657
# Relevance: Array, Hash Table, Bit Manipulation

from typing import List

class Solution:

    # Logic: Hash Set (O(N) Time, O(N) Space)
    # 1. Initialize an empty set `seen` to keep track of elements encountered.
    # 2. Iterate through arrays `A` and `B` simultaneously from index `0` to `n-1`.
    # 3. For each index `i`:
    #    - If `A[i] == B[i]`, they match directly, increment the common prefix count (`cnt`) by 1.
    #    - If `A[i]` is in `seen`, it has already appeared in `B`, increment `cnt` by 1.
    #    - If `B[i]` is in `seen`, it has already appeared in `A`, increment `cnt` by 1.
    #    - Add both `A[i]` and `B[i]` to `seen`.
    #    - Append the current prefix common count `cnt` to the result.
    # 4. Return the result list.
    #
    # Dry run for A = [1, 3, 2, 4], B = [3, 1, 2, 4]:
    # 1. i = 0: A[0]=1, B[0]=3. 
    #    - Neither is in `seen`. seen = {1, 3}, res = [0], cnt = 0.
    # 2. i = 1: A[1]=3, B[1]=1. 
    #    - Both `A[1]` and `B[1]` are in `seen`. cnt increases by 2.
    #    - seen = {1, 3}, res = [0, 2], cnt = 2.
    # 3. i = 2: A[2]=2, B[2]=2.
    #    - A[2] == B[2] is True. cnt increases by 1.
    #    - seen = {1, 2, 3}, res = [0, 2, 3], cnt = 3.
    # 4. i = 3: A[3]=4, B[3]=4.
    #    - A[3] == B[3] is True. cnt increases by 1.
    #    - seen = {1, 2, 3, 4}, res = [0, 2, 3, 4], cnt = 4.
    # Returns [0, 2, 3, 4].

    def findThePrefixCommonArray(self, A: List[int], B: List[int]) -> List[int]:
        seen = set()
        cnt = 0
        res = []
        for i in range(len(A)):
            if A[i] == B[i]:
                cnt+=1
            if A[i] in seen:
                cnt +=1
            if B[i] in seen:
                cnt +=1
            seen.add(A[i])
            seen.add(B[i])
            res.append(cnt)
        return res

if __name__ == "__main__":
    sol = Solution()

    # Example 1: A = [1, 3, 2, 4], B = [3, 1, 2, 4] -> Expected: [0, 2, 3, 4]
    A1 = [1, 3, 2, 4]
    B1 = [3, 1, 2, 4]
    print(f"Result for {A1} and {B1}: {sol.findThePrefixCommonArray(A1, B1)}")

    # Example 2: A = [2, 3, 1], B = [3, 1, 2] -> Expected: [0, 1, 3]
    A2 = [2, 3, 1]
    B2 = [3, 1, 2]
    print(f"Result for {A2} and {B2}: {sol.findThePrefixCommonArray(A2, B2)}")
