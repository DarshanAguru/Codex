# Problem: https://leetcode.com/problems/maximum-amount-of-money-robot-can-earn/
# Leetcode Problem No: 3418
# Relevance: Dynamic Programming, Array, Matrix

from typing import List
from math import inf

class Solution:

    # Logic: 3D Grid Dynamic Programming
    # We want to find the maximum money a robot can collect while moving from (0,0) to (m-1, n-1).
    # The robot can "ignore" up to 2 negative coin cells (treating their cost as 0).
    # State: dp[i][j][k] = maximum amount at cell (i, j) having used k skips.
    # Base cases: First cell (dp[0][0][k]) and first row/column edges.
    # Transition:
    # 1. Without using a skip at (i,j): max(dp[i-1][j][k], dp[i][j-1][k]) + coins[i][j]
    # 2. Using a skip at (i,j) (only if k > 0): max(dp[i-1][j][k-1], dp[i][j-1][k-1])
    # Note: If the coin at (i,j) is positive, skips are naturally not used as they result in a smaller value.
    
    def maximumAmount(self, coins: List[List[int]]) -> int:
        m, n = len(coins), len(coins[0])
        dp = [[[-inf] * 3 for _ in range(n)] for _ in range(m)]

        # Starting cell initialization
        dp[0][0][0] = coins[0][0]
        for k in range(1, 3):
            dp[0][0][k] = max(coins[0][0], 0)

        # First row initialization
        for j in range(1, n):
            dp[0][j][0] = dp[0][j - 1][0] + coins[0][j]
            x = max(coins[0][j], 0)
            for k in range(1, 3):
                dp[0][j][k] = max(
                    dp[0][j - 1][k] + coins[0][j], dp[0][j - 1][k - 1] + x
                )

        # First column initialization
        for i in range(1, m):
            dp[i][0][0] = dp[i - 1][0][0] + coins[i][0]
            x = max(coins[i][0], 0)
            for k in range(1, 3):
                dp[i][0][k] = max(
                    dp[i - 1][0][k] + coins[i][0], dp[i - 1][0][k - 1] + x
                )

        # Fill the rest of the DP table
        for i in range(1, m):
            for j in range(1, n):
                x = coins[i][j]
                
                # Using 0 skips
                dp[i][j][0] = max(dp[i - 1][j][0], dp[i][j - 1][0]) + x
                
                # Using 1 skip
                # Either we don't use a skip at (i,j) OR we use our 1 allowed skip at (i,j)
                dp[i][j][1] = max(
                    dp[i - 1][j][1] + x,   # from top, didn't skip (i,j)
                    dp[i][j - 1][1] + x,   # from left, didn't skip (i,j)
                    dp[i - 1][j][0],       # from top, used the skip at (i,j)
                    dp[i][j - 1][0]        # from left, used the skip at (i,j)
                )

                # Using 2 skips
                # Either we don't use a skip at (i,j) (having used 2 already)
                # OR we use our last allowed skip at (i,j) (having used 1 already)
                dp[i][j][2] = max(
                    dp[i - 1][j][2] + x,   # from top, didn't skip (i,j)
                    dp[i][j - 1][2] + x,   # from left, didn't skip (i,j)
                    dp[i - 1][j][1],       # from top, used the skip at (i,j)
                    dp[i][j - 1][1]        # from left, used the skip at (i,j)
                )

        return int(dp[m - 1][n - 1][2]) # ensure result is integer

    sol = Solution()
    
    # Example: [[0,1,-1],[1,-2,1]] -> Expected Result: 3
    coins = [[0,1,-1],[1,-2,1]]
    print(f"Result: {sol.maximumAmount(coins)}")
