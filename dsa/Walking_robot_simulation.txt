from typing import List

class Solution:

    # Logic: Simulation with Direction Tracking and Coordinate Hashing
    # 1. Convert obstacles list to a set of tuples for O(1) lookup.
    # 2. Initialize current position (0, 0), current direction (North), and max_dist (0).
    # 3. Define direction vectors: North (0,1), East (1,0), South (0,-1), West (-1,0).
    # 4. Iterate through each command:
    #    - If -1: Turn right (curr_dir + 1) % 4.
    #    - If -2: Turn left (curr_dir + 3) % 4.
    #    - If positive (k): Move k steps forward one by one.
    #      - For each step, check if the next position is an obstacle.
    #      - If yes, stop moving for this command.
    #      - If no, update (x, y).
    # 5. After each command, update max_dist = max(max_dist, x^2 + y^2).
    #
    # Dry run for commands = [4,-1,3], obstacles = []:
    # 1. Init: x=0, y=0, curr_dir=0 (North), max_dist=0.
    # 2. command=4: move North 4 steps. x=0, y=4. max_dist=16.
    # 3. command=-1: Turn right. curr_dir=1 (East).
    # 4. command=3: move East 3 steps. x=3, y=4. max_dist=max(16, 3^2 + 4^2) = 25.
    # 5. Final: 25.

    def robotSim(self, commands: List[int], obstacles: List[List[int]]) -> int:
        obs_set = set([(x,y) for x,y in obstacles])
        dirs = [(0,1), (1,0), (0,-1), (-1,0)] # N, E, S, W
        x, y = 0, 0
        max_dist = 0
        curr_dir = 0

        for command in commands:
            if command == -1:
                curr_dir = (curr_dir + 1) % 4
            elif command == -2:
                curr_dir = (curr_dir + 3) % 4
            else:
                dx, dy = dirs[curr_dir]
                for _ in range(command):
                    new_x, new_y = x + dx, y + dy
                    if (new_x, new_y) in obs_set:
                        break
                    x, y = new_x, new_y
                max_dist = max(max_dist, x*x + y*y)
        
        return max_dist

if __name__ == "__main__":
    sol = Solution()
    
    # Example 1: commands = [4,-1,3], obstacles = []
    # Output: 25
    print(f"Example 1: {sol.robotSim([4,-1,3], [])}") 

    # Example 2: commands = [4,-1,4,-2,4], obstacles = [[2,4]]
    # Output: 65
    print(f"Example 2: {sol.robotSim([4,-1,4,-2,4], [[2,4]])}")
