# Problem: https://leetcode.com/problems/minimum-absolute-distance-between-mirror-pairs/
# Leetcode Problem No: 3761
# Relevance: Array, Hash Table, String

from typing import List

class Solution:

    # Logic: Hash Map with Digit-Reversal
    # A "mirror pair" is a pair (nums[i], nums[j]) where one is the digit-reversal of the other.
    # 1. Iterate through 'nums', maintaining a 'seen' map: {number -> most recent index}.
    # 2. For each number 'num', check if it already exists in 'seen'.
    #    - If yes, it means its mirror (the reverse of some earlier number) matches 'num',
    #      so (seen[num], i) forms a mirror pair. Update res with abs(i - seen[num]).
    # 3. After checking, store the digit-reversal of 'num' in 'seen' mapped to index 'i'.
    #    This pre-populates the map so future numbers can match against it.
    # 4. If no mirror pair is ever found, return -1.
    #
    # Example: nums = [13, 31, 12, 21]
    # i=0: num=13. Is 13 in seen? No. seen[reverse(13)=31] = 0. seen={31:0}
    # i=1: num=31. Is 31 in seen? Yes (idx=0). res=min(inf, |1-0|)=1. seen={31:0, reverse(31)=13:1}
    # i=2: num=12. Is 12 in seen? No. seen[reverse(12)=21] = 2.
    # i=3: num=21. Is 21 in seen? Yes (idx=2). res=min(1, |3-2|)=1. 
    # Result: 1
    def minMirrorPairDistance(self, nums: List[int]) -> int:
        seen = {}
        res = float('inf')

        def reverse(n: int) -> int:
            return int(str(n)[::-1])

        for i, num in enumerate(nums):
            if num in seen:
                res = min(res, abs(i - seen[num]))

            seen[reverse(num)] = i

        return res if res != float('inf') else -1

    print(f"Example 1 Result: {sol.minMirrorPairDistance([13, 31, 12, 21])}") # Expected: 1

    # Example 2: nums = [1, 2, 3]
    print(f"Example 2 Result: {sol.minMirrorPairDistance([1, 2, 3])}") # Expected: -1

    # Example 3: nums = [11, 23, 32, 11]
    print(f"Example 3 Result: {sol.minMirrorPairDistance([11, 23, 32, 11])}") # Expected: 1
