// Date: 27/03/2026
// Problem: https://leetcode.com/problems/check-if-matrix-is-similar-after-cyclic-shifts/
// Leetcode Problem No: 2946
// Relevance: Matrix, Simulation, Modulo Arithmetic

import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[][] mat = { { 1, 2, 1, 2 }, { 5, 5, 5, 5 }, { 6, 3, 6, 3 } };
    int k = 2;

    // Logic:
    // Each row is cyclically shifted:
    // - Even index rows (0-based): shift RIGHT by k
    // - Odd index rows: shift LEFT by k
    //
    // Instead of actually shifting, we simulate the final position:
    // For each element mat[i][j], compute where it should land after shift.
    //
    // Even row (i % 2 == 0):
    //   newCol = (j + k) % n   → right shift
    //
    // Odd row:
    //   newCol = (j - k + n) % n → left shift (adding n avoids negative index)
    //
    // If after shift, the value at original index and shifted index differ,
    // the matrix is NOT similar.
    //
    // Dry Run:
    // mat = [
    //  [1,2,1,2],
    //  [5,5,5,5],
    //  [6,3,6,3]
    // ], k = 2
    //
    // Row 0 (even): right shift by 2 → [1,2,1,2] (same)
    // Row 1 (odd): left shift by 2 → [5,5,5,5] (same)
    // Row 2 (even): right shift by 2 → [6,3,6,3] (same)
    //
    // All rows match → true

    int m = mat.length;
    int n = mat[0].length;

    k %= n;

    boolean isSimilar = true;

    for (int i = 0; i < m; i++) {
      for (int j = 0; j < n; j++) {
        int newCol;

        if (i % 2 == 0) {
          newCol = (j + k) % n;
        } else {
          newCol = (j - k + n) % n;
        }

        if (mat[i][j] != mat[i][newCol]) {
          isSimilar = false;
          break;
        }
      }
      if (!isSimilar) break;
    }

    System.out.println(isSimilar);
  }
}
