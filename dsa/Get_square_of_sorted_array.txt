// Problem: https://leetcode.com/problems/squares-of-a-sorted-array/
// Leetcode Problem No: 977
// Relevance: Array, Sorting, Two Pointers

import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[] arr = { -4, -1, 0, 3, 10 };
    int[] res = getSquaresOfSortedArray(arr);
    System.out.println(Arrays.toString(res));
  }

  // Logic: Two Pointers (Outside-In)
  // The given array is sorted (e.g., contains negative numbers followed by positive numbers).
  // When we square the elements, the largest squares will be at the ends of the array
  // (either large negative numbers at the start or large positive numbers at the end).
  // So, we use two pointers (`left` and `right`) and compare their absolute (or squared) values.
  // We write the larger square to the *end* of our result array, and move the corresponding pointer inward.
  //
  // Dry run for arr = [-4, -1, 0, 3, 10]:
  // res array of size 5. i=4. left=0 (-4), right=4 (10).
  // i=4: |10| > |-4|, res[4] = 100. right=3.
  // i=3: |3| < |-4|, res[3] = 16. left=1.
  // i=2: |3| > |-1|, res[2] = 9. right=2.
  // i=1: |0| < |-1|, res[1] = 1. left=2.
  // i=0: res[0] = 0.
  // Final array: [0, 1, 9, 16, 100].
  static int[] getSquaresOfSortedArray(int[] arr) {
    int n = arr.length;
    int[] res = new int[n];
    int left = 0;
    int right = n - 1;

    for (int i = n - 1; i >= 0; i--) {
      if (Math.abs(arr[left]) > Math.abs(arr[right])) {
        res[i] = arr[left] * arr[left];
        left++;
      } else {
        res[i] = arr[right] * arr[right];
        right--;
      }
    }
    return res;
  }
}
