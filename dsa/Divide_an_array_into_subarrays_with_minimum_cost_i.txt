// Problem: https://leetcode.com/problems/divide-an-array-into-subarrays-with-minimum-cost-i/
// Leetcode Problem No: 3010
// Relevance: Array, Sorting, Enumeration

import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[] nums = { 1, 2, 3, 12 };

    int res = minimumCost(nums);

    System.out.println(res);
  }

  public static int minimumCost(int[] nums) {
    int INF = Integer.MAX_VALUE;
    int fixed = nums[0];
    int min1 = INF;
    int min2 = INF;

    for (int i = 1; i < nums.length; i++) {
      if (nums[i] < min1) {
        min2 = min1;
        min1 = nums[i];
      } else if (nums[i] < min2) {
        min2 = nums[i];
      }
    }

    return fixed + min1 + min2;
  }
}
