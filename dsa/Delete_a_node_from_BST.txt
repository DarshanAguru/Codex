// Problem: https://leetcode.com/problems/delete-node-in-a-bst/

import java.util.*;
import java.lang.*;

class Node {
    public Node left, right;
    public int val;
    Node(int val) {
        this.val = val;
    }
}

class Main {
    public static void main(String[] args) {
        Node root = new Node(5);
        root.left = new Node(3);
        root.left.left = new Node(2);
        root.left.right = new Node(4);
        root.right = new Node(6);
        root.right.right = new Node(7);
        
        int target = 3;
        root = deleteNodeFromBST(root, target);
        printInorder(root);
    }
    
    static void printInorder(Node node) {
        if (node == null) return;
        printInorder(node.left);
        System.out.print(node.val + " ");
        printInorder(node.right);
    }
    
    static Node deleteNodeFromBST(Node node, int target) {
        if (node == null) return null;
        
        if (target < node.val) {
            node.left = deleteNodeFromBST(node.left, target);
        } else if (target > node.val) {
            node.right = deleteNodeFromBST(node.right, target);
        } else {
            // Node found
            if (node.left == null) return node.right;
            if (node.right == null) return node.left;
            
            // Node has two children
            Node next = getMin(node.right);
            node.val = next.val;
            node.right = deleteNodeFromBST(node.right, next.val);
        }
        return node;
    }
    
    static Node getMin(Node node) {
        while (node.left != null) {
            node = node.left;
        }
        return node;
    }
}
