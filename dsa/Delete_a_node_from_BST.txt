// Relevance: Tree, BST

import java.lang.*;
import java.util.*;

class Node {

  public Node left, right;
  public int val;

  Node(int val) {
    this.val = val;
  }
}

class Main {

  public static void main(String[] args) {
    Node root = new Node(5);
    root.left = new Node(3);
    root.left.left = new Node(2);
    root.left.right = new Node(4);
    root.right = new Node(6);
    root.right.right = new Node(7);

    int target = 3;
    root = deleteNodeFromBST(root, target);
    printInorder(root);
  }

  static void printInorder(Node node) {
    if (node == null) return;
    printInorder(node.left);
    System.out.print(node.val + " ");
    printInorder(node.right);
  }

  // Logic: BST Deletion Cases
  // To delete a node from a BST, we first find it using BST properties (left for smaller, right for larger).
  // Once found, there are 3 cases:
  // Case 1: Node is a leaf (both children null) -> Return null to parent.
  // Case 2: Node has one child -> Return the non-null child to the parent.
  // Case 3: Node has two children -> Find the inorder successor (smallest node in the right subtree).
  //         Replace the node's value with the successor's value, and recursively delete the successor
  //         from the right subtree.
  //
  // Dry run for root=[5,3,6,2,4,null,7], target=3:
  // 3 < 5, go left to node(3). target == 3. Found it.
  // Node(3) has two children (2 and 4).
  // Inorder successor is getMin(node.right) -> getMin(node(4)) -> returns 4.
  // Replace node(3).val with 4. Node is now 4.
  // Call deleteNodeFromBST(node.right (which is 4), target=4).
  // Inside recursive call: target==4. Found it. It's a leaf (both children null). Returns null.
  // So node(4).right becomes null.
  // Resulting subtree: 4 -> L:2, R:null.
  static Node deleteNodeFromBST(Node node, int target) {
    if (node == null) return null;

    if (target < node.val) {
      node.left = deleteNodeFromBST(node.left, target);
    } else if (target > node.val) {
      node.right = deleteNodeFromBST(node.right, target);
    } else {
      // Node found
      if (node.left == null) return node.right;
      if (node.right == null) return node.left;

      // Node has two children
      Node next = getMin(node.right);
      node.val = next.val;
      node.right = deleteNodeFromBST(node.right, next.val);
    }
    return node;
  }

  static Node getMin(Node node) {
    while (node.left != null) {
      node = node.left;
    }
    return node;
  }
}
