# Problem: https://leetcode.com/problems/number-of-ways-to-assign-edge-weights-i/
# Leetcode Problem No: 3558
# Relevance: Tree, Depth-First Search, Combinatorics

from collections import defaultdict
from typing import List

class Solution:

    # Logic: Tree Depth-First Search and Subset Combinatorics (O(N) Time, O(N) Space)
    # 1. We are given an undirected tree rooted at node 1.
    # 2. We want to find the number of ways to assign weights (1 or 2) to the edges along a path of maximum length L (from node 1 to a leaf node at maximum depth) such that the sum of the weights on the path is odd.
    # 3. Sum of weights = 1 * o + 2 * (L - o) = 2L - o, where `o` is the number of edges assigned weight 1.
    # 4. The sum is odd if and only if `o` (the number of odd weights) is odd.
    # 5. By combinatorics, the number of ways to choose an odd number of items from a set of size L is exactly 2^(L-1).
    # 6. We find the maximum path length (number of edges L) from root 1 using a simple DFS traversal.
    # 7. Finally, we calculate 2^(L - 1) % (10^9 + 7).
    #
    # Dry run for edges = [[1, 2], [1, 3], [2, 4]]:
    # 1. Build adjacency list: tree = {1: [2, 3], 2: [1, 4], 3: [1], 4: [2]}
    # 2. Call dfs(1, 0):
    #    - For neighbor 2 (dfs(2, 1)):
    #      - For neighbor 4 (dfs(4, 2)):
    #        - No unvisited neighbors. Returns 0.
    #      - Returns 1 + 0 = 1.
    #    - For neighbor 3 (dfs(3, 1)):
    #      - No unvisited neighbors. Returns 0.
    #    - dfs(1, 0) returns max(1 + 1, 1 + 0) = 2.
    # 3. max_depth = 2.
    # 4. res = pow(2, 2 - 1, 10^9 + 7) = 2.
    # 5. Returns 2.

    def assignEdgeWeights(self, edges: List[List[int]]) -> int:
        MOD = 10**9 + 7
        tree = defaultdict(list)
        for u, v in edges:
            tree[u].append(v)
            tree[v].append(u)
        
        def dfs(n, p):
            d = 0
            for e in tree[n]:
                if e != p:
                    d = max(d, 1 + dfs(e, n))
            return d

        max_depth = dfs(1, 0)
        res = pow(2, max_depth - 1, MOD)
        return res

if __name__ == "__main__":
    sol = Solution()

    # Example 1: edges = [[1, 2], [1, 3], [2, 4]] -> Expected: 2
    edges1 = [[1, 2], [1, 3], [2, 4]]
    print(f"Result: {sol.assignEdgeWeights(edges1)}")

    # Example 2: edges = [[1, 2], [1, 3]] -> Expected: 1
    edges2 = [[1, 2], [1, 3]]
    print(f"Result: {sol.assignEdgeWeights(edges2)}")
