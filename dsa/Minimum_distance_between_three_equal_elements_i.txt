# Leetcode Problem No: 3740
# Relevance: Array, Hash Table

import math
from collections import defaultdict
from typing import List

class Solution:

    # Logic: Index Mapping & Minimum Gap
    # 1. Use a hash map (defaultdict) to store all indices for each unique element in the array.
    # 2. Iterate through each list of indices in the map.
    # 3. For any list with at least 3 indices, calculate the distance between the 1st and 3rd 
    #    occurrence (v[i] and v[i-2]) across all sliding windows of size 3.
    # 4. The problem defines the distance as 2 * (v[i] - v[i-2]).
    # 5. Keep track of the minimum such distance found across all elements.
    # 6. Return the minimum distance, or -1 if no element appears at least three times.
    #
    # Dry run for nums = [1, 2, 1, 3, 1, 2, 2]:
    # 1. Index Map:
    #    - 1: [0, 2, 4]
    #    - 2: [1, 5, 6]
    #    - 3: [3]
    # 2. Check 1: indices [0, 2, 4]. i=2: dist = 2 * (4 - 0) = 8.
    # 3. Check 2: indices [1, 5, 6]. i=2: dist = 2 * (6 - 1) = 10.
    # 4. Check 3: indices [3]. (Less than 3 elements).
    # 5. Min distance found: 8.
    def minimumDistance(self, nums: List[int]) -> int:
        num_idx_map = defaultdict(list)
        for (idx, num) in enumerate(nums):
            num_idx_map[num].append(idx)

        res = math.inf
        for k, v in num_idx_map.items():
            for i in range(2, len(v)):
                dist = 2 * (v[i] - v[i-2])
                res = min(res, dist)
                
        return -1 if res == math.inf else int(res)

if __name__ == "__main__":
    sol = Solution()
    
    # Example 1
    nums1 = [1, 2, 1, 3, 1, 2, 2]
    print(f"Example 1: {sol.minimumDistance(nums1)}") # Expected: 8
    
    # Example 2
    nums2 = [1, 1, 1, 1]
    # v: [0, 1, 2, 3]
    # i=2: 2*(2-0) = 4
    # i=3: 2*(3-1) = 4
    print(f"Example 2: {sol.minimumDistance(nums2)}") # Expected: 4
    
    # Example 3
    nums3 = [1, 2, 3, 4]
    print(f"Example 3: {sol.minimumDistance(nums3)}") # Expected: -1
