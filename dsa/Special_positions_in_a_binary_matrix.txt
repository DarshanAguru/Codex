import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[][] mat = { { 1, 0, 0 }, { 0, 0, 1 }, { 1, 0, 0 } };
    System.out.println(numSpecial(mat));
  }

  static int numSpecial(int[][] mat) {
    int m = mat.length;
    int n = mat[0].length;
    int[] row = new int[m];
    int[] col = new int[n];
    int count = 0;

    for (int i = 0; i < m; i++) {
      for (int j = 0; j < n; j++) {
        if (mat[i][j] == 1) {
          row[i]++;
          col[j]++;
        }
      }
    }

    for (int i = 0; i < m; i++) {
      for (int j = 0; j < n; j++) {
        if (mat[i][j] == 1 && row[i] == 1 && col[j] == 1) {
          count++;
        }
      }
    }

    return count;
  }
}
