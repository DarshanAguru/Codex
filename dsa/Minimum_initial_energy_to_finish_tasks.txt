# Problem: https://leetcode.com/problems/minimum-initial-energy-to-finish-tasks/
# Leetcode Problem No: 1665
# Relevance: Greedy, Array, Sorting

from typing import List

class Solution:

    # Logic: Greedy Sorting by Difference (O(N log N))
    # Each task has [actual_cost, minimum_requirement].
    # The key observation is that tasks with a larger "cushion" (minimum - actual) 
    # should be performed first. This allows us to use the extra energy (the cushion) 
    # to meet the minimum requirements of later tasks.
    # 1. Sort tasks by the difference `minimum - actual` in descending order.
    # 2. Iterate through sorted tasks:
    #    - If current energy `left` is less than `task[1]`, we need to increase 
    #      our starting energy.
    #    - Update total initial energy `res` and current energy `left`.
    #    - Subtract `task[0]` from `left`.
    #
    # Dry run for tasks = [[1, 3], [2, 4], [10, 11]]:
    # Sorted by (min-act): [[10, 11] (diff 1), [2, 4] (diff 2), [1, 3] (diff 2)]
    # Wait, diffs are 1, 2, 2. Sort desc: [[2, 4], [1, 3], [10, 11]] (or swap [2,4] and [1,3])
    # 1. [2, 4]: left=0. left < 4. Need 4. res=4, left=4. After task: left = 4-2 = 2.
    # 2. [1, 3]: left=2. left < 3. Need 1 more. res=5, left=3. After task: left = 3-1 = 2.
    # 3. [10, 11]: left=2. left < 11. Need 9 more. res=14, left=11. After task: left = 11-10 = 1.
    # Output: 14.

    def minimumEffort(self, tasks: List[List[int]]) -> int:
        # Sort by the cushion (minimum - actual) in descending order
        tasks.sort(key=lambda x: x[1] - x[0], reverse=True)
        
        res = 0  # Total initial energy needed
        left = 0 # Current energy available
        
        for actual, minimum in tasks:
            if left < minimum:
                # We need more energy to start this task
                diff = minimum - left
                res += diff
                left += diff
            # Consume the actual energy for the task
            left -= actual
            
        return res

if __name__ == "__main__":
    sol = Solution()

    # Example 1: tasks = [[1,2],[2,4],[4,8]] -> Expected: 8
    tasks1 = [[1, 2], [2, 4], [4, 8]]
    print(f"Result for {tasks1}: {sol.minimumEffort(tasks1)}")

    # Example 2: tasks = [[1,3],[2,4],[10,11],[10,12],[8,9]] -> Expected: 32
    tasks2 = [[1, 3], [2, 4], [10, 11], [10, 12], [8, 9]]
    print(f"Result for {tasks2}: {sol.minimumEffort(tasks2)}")

    # Example 3: tasks = [[1,7],[2,8],[3,9],[4,10],[5,11],[6,12]] -> Expected: 27
    tasks3 = [[1, 7], [2, 8], [3, 9], [4, 10], [5, 11], [6, 12]]
    print(f"Result for {tasks3}: {sol.minimumEffort(tasks3)}")
