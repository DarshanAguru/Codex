# Relevance: Graph, Depth-First Search, Breadth-First Search, Union Find

from typing import List

class DSU:
    def __init__(self, n):
        n += 1
        self.parent = [i for i in range(n)]
        self.rank = [0] * n
    
    def find(self, x):
        if self.parent[x] == x:
            return x
        self.parent[x] = self.find(self.parent[x])
        return self.parent[x]
    
    def union(self, x, y):
        rx = self.find(x)
        ry = self.find(y)

        if rx == ry:
            return
        
        if self.rank[rx] < self.rank[ry]:
            self.parent[rx] = ry
        elif self.rank[rx] > self.rank[ry]:
            self.parent[ry] = rx
        else:
            self.parent[ry] = rx
            self.rank[rx] += 1

class Solution:

    # Logic: Union-Find (DSU) (O(V + E * alpha(V)) Time, O(V) Space)
    # 1. The problem asks for the minimum score (minimum edge weight) of a path from city 1 to city n.
    #    Since we can traverse edges multiple times, any edge in the same connected component as city 1
    #    can be part of the path.
    # 2. We can use a Disjoint Set Union (DSU) data structure to group cities connected by roads.
    # 3. First, we union the endpoints of all roads.
    # 4. After grouping, the connected component containing city 1 will also contain city n (since a valid path is guaranteed to exist).
    # 5. We identify the representative/root node of city 1's component: `start = dsu.find(1)`.
    # 6. Finally, we iterate through all the roads. For any road `(u, v, val)` where `u` (or `v`) is in the same component
    #    as city 1 (`dsu.find(u) == start`), we update our minimum score result `res = min(res, val)`.
    # 7. We return `res`.
    #
    # Dry run for n = 4, roads = [[1, 2, 9], [2, 3, 6], [2, 4, 5], [1, 4, 7]]:
    # DSU initialized for 4 cities.
    #
    # Step 1: Union roads
    # - Road [1, 2, 9]: union(1, 2) -> parent: {1: 1, 2: 1, 3: 3, 4: 4}
    # - Road [2, 3, 6]: union(2, 3) -> parent: {1: 1, 2: 1, 3: 1, 4: 4}
    # - Road [2, 4, 5]: union(2, 4) -> parent: {1: 1, 2: 1, 3: 1, 4: 1}
    # - Road [1, 4, 7]: union(1, 4) -> already unioned
    # All cities are now in component 1. Root of city 1 is `start = 1`.
    #
    # Step 2: Find minimum edge weight in component 1
    # - Road [1, 2, 9]: find(1) == 1. res = min(inf, 9) = 9
    # - Road [2, 3, 6]: find(2) == 1. res = min(9, 6) = 6
    # - Road [2, 4, 5]: find(2) == 1. res = min(6, 5) = 5
    # - Road [1, 4, 7]: find(1) == 1. res = min(5, 7) = 5
    # Result: 5.

    def minScore(self, n: int, roads: List[List[int]]) -> int:
        dsu = DSU(n)
        
        for u, v, val in roads:
            dsu.union(u, v)
        
        start = dsu.find(1)
        res = float("inf")
        
        for u, v, val in roads:
            if dsu.find(u) == start:
                res = min(res, val)
        return res

if __name__ == "__main__":
    sol = Solution()

    # Example 1: n = 4, roads = [[1,2,9],[2,3,6],[2,4,5],[1,4,7]] -> Expected: 5
    n1 = 4
    roads1 = [[1, 2, 9], [2, 3, 6], [2, 4, 5], [1, 4, 7]]
    print(f"Result for example 1: {sol.minScore(n1, roads1)}")

    # Example 2: n = 4, roads = [[1,2,24],[1,4,37],[2,3,18],[3,4,12]] -> Expected: 12
    n2 = 4
    roads2 = [[1, 2, 24], [1, 4, 37], [2, 3, 18], [3, 4, 12]]
    print(f"Result for example 2: {sol.minScore(n2, roads2)}")
