// Problem: https://leetcode.com/problems/sudoku-solver/

import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    char[][] board = fillBoard();

    System.out.println("INPUT:");
    printBoard(board);

    if (solveSudoku(board)) {
      System.out.println("OUTPUT:");
      printBoard(board);
    } else {
      System.out.println("No solution exists");
    }
  }

  static boolean solveSudoku(char[][] board) {
    for (int i = 0; i < 9; i++) {
      for (int j = 0; j < 9; j++) {
        if (board[i][j] == '.') {
          for (char c = '1'; c <= '9'; c++) {
            if (validate(board, i, j, c)) {
              board[i][j] = c;

              if (solveSudoku(board)) {
                return true;
              } else {
                board[i][j] = '.'; // Backtrack
              }
            }
          }
          return false;
        }
      }
    }
    return true;
  }

  static boolean validate(char[][] board, int r, int c, char x) {
    for (int i = 0; i < 9; i++) {
      // Check Row
      if (board[r][i] == x) return false;

      // Check Column
      if (board[i][c] == x) return false;

      // Check 3x3 Subgrid
      if (board[3 * (r / 3) + i / 3][3 * (c / 3) + (i % 3)] == x) return false;
    }
    return true;
  }

  static char[][] fillBoard() {
    return new char[][] {
      { '9', '5', '7', '.', '1', '3', '.', '8', '4' },
      { '4', '8', '3', '.', '5', '7', '1', '.', '6' },
      { '.', '1', '2', '.', '4', '9', '5', '3', '7' },
      { '1', '7', '.', '3', '.', '4', '9', '.', '2' },
      { '5', '.', '4', '9', '7', '.', '3', '6', '.' },
      { '3', '.', '9', '5', '.', '8', '7', '.', '1' },
      { '8', '4', '5', '7', '9', '.', '6', '1', '3' },
      { '.', '9', '1', '.', '3', '6', '.', '7', '5' },
      { '7', '.', '6', '1', '8', '5', '4', '.', '9' },
    };
  }

  static void printBoard(char[][] board) {
    for (char[] row : board) {
      System.out.println(Arrays.toString(row));
    }
  }
}
