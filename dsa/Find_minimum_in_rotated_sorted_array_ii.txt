# Problem: https://leetcode.com/problems/find-minimum-in-rotated-sorted-array-ii/
# Leetcode Problem No: 154
# Relevance: Array, Binary Search

from typing import List
from bisect import bisect_left

class Solution:

    # Logic: Binary Search with Custom Key (O(log N) Average, O(N) Worst Case Time, O(1) Space)
    # 1. To handle duplicates at boundaries, pop elements from the end of the array that are equal to `nums[0]`.
    # 2. Once duplicates at the end are removed, the rotated sorted array has two parts:
    #    - Left part: Elements that are strictly greater than `nums[-1]`.
    #    - Right part: Elements that are less than or equal to `nums[-1]`.
    # 3. If we map each element `x` to the boolean condition `x <= nums[-1]`, we get a sequence of:
    #    `[False, False, ..., False, True, True, ..., True]`.
    # 4. The first occurrence of `True` in this mapped sequence is the minimum element.
    # 5. Use `bisect_left` with `key = lambda x: x <= nums[-1]` to binary search for `True`.
    #
    # Dry run for nums = [2, 2, 2, 0, 1]:
    # 1. nums[-1] = 1, nums[0] = 2. Since 1 != 2, the duplicate removal loop doesn't run.
    # 2. Key mapping `x <= 1` on nums = [2, 2, 2, 0, 1]:
    #    - 2 <= 1 -> False
    #    - 2 <= 1 -> False
    #    - 2 <= 1 -> False
    #    - 0 <= 1 -> True
    #    - 1 <= 1 -> True
    #    Mapped sequence: [False, False, False, True, True]
    # 3. bisect_left(..., True) finds the first True at index 3.
    # 4. Returns nums[3] = 0.

    def findMin(self, nums: List[int]) -> int:
        while len(nums) > 1 and nums[-1] == nums[0]:
            nums.pop()
        return nums[bisect_left(nums, True, key = lambda x: x <= nums[-1])]

if __name__ == "__main__":
    sol = Solution()

    # Example 1: nums = [1, 3, 5] -> Expected: 1
    nums1 = [1, 3, 5]
    print(f"Result for {nums1}: {sol.findMin(nums1)}")

    # Example 2: nums = [2, 2, 2, 0, 1] -> Expected: 0
    nums2 = [2, 2, 2, 0, 1]
    print(f"Result for {nums2}: {sol.findMin(nums2)}")

    # Example 3: nums = [2, 2, 2, 0, 2] -> Expected: 0
    nums3 = [2, 2, 2, 0, 2]
    print(f"Result for {nums3}: {sol.findMin(nums3)}")
