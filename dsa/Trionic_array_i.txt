import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[] nums = { 1, 2, 3, 2, 4, 5 };
    System.out.println(isTrionic(nums));
  }

  // Logic: Array Simulation / Peak-Valley Counting
  // A trionic array is one that changes direction exactly twice.
  // We need to count the number of "turns" (peaks or valleys).
  // First, check if `nums[0] >= nums[1]`. If it is, return false (must start strictly increasing).
  // Next, iterate from index 2 onward.
  // If `nums[i-1] == nums[i]`, there's a flat plateau, return false.
  // A turn occurs when `(nums[i-2] - nums[i-1]) * (nums[i-1] - nums[i]) < 0`.
  // We start `count` at 1 (the initial increasing sequence).
  // Return true if `count` exactly equals 3 (Increase -> Decrease -> Increase).
  //
  // Dry run for nums = [1, 2, 3, 2, 4, 5]:
  // nums[0](1) >= nums[1](2) -> False.
  // count = 1.
  // i=2: 1-2 = -1, 2-3 = -1. Product = 1 > 0. (No turn)
  // i=3: 2-3 = -1, 3-2 = 1. Product = -1 < 0. Turn! count = 2.
  // i=4: 3-2 = 1, 2-4 = -2. Product = -2 < 0. Turn! count = 3.
  // i=5: 2-4 = -2, 4-5 = -1. Product = 2 > 0.
  // count = 3 -> Returns true!
  public static boolean isTrionic(int[] nums) {
    int n = nums.length;
    if (nums[0] >= nums[1]) {
      return false;
    }
    int count = 1;
    for (int i = 2; i < n; i++) {
      if (nums[i - 1] == nums[i]) {
        return false;
      }
      if ((nums[i - 2] - nums[i - 1]) * (nums[i - 1] - nums[i]) < 0) {
        count++;
      }
    }
    return count == 3;
  }
}
