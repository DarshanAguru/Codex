# Problem: https://leetcode.com/problems/check-if-array-is-good/
# Leetcode Problem No: 2784
# Relevance: Array, Counting, Hash Table

from typing import List

class Solution:

    # Logic: Tracking Visited Elements (O(N) Time, O(N) Space)
    # 1. Identify `n` as `len(nums) - 1`.
    # 2. Use a set `vis` to track visited numbers and a boolean `rep` for the duplicate `n`.
    # 3. Iterate through `nums`:
    #    - If `num > n`, it's invalid.
    #    - If `num` is already in `vis`:
    #        - If `num < n` (should be unique) or `rep` is already True, it's invalid.
    #        - Otherwise, mark `rep = True`.
    #    - Add `num` to `vis`.
    # 4. Return True if all checks pass.
    #
    # Dry run for nums = [1, 3, 3, 2]:
    # n = 3. 
    # 1. num = 1: vis = {1}
    # 2. num = 3: vis = {1, 3}
    # 3. num = 3: 3 in vis, 3 is not < 3, rep = False -> rep = True
    # 4. num = 2: vis = {1, 3, 2}
    # Returns True.

    def isGood(self, nums: List[int]) -> bool:
        n = len(nums) - 1
        vis = set()
        rep = False
        for num in nums:
            if num > n:
                return False
            
            if num in vis:
                if num < n or rep:
                    return False
                rep = True
                continue
            
            vis.add(num)
        return True

if __name__ == "__main__":
    sol = Solution()

    # Example 1: nums = [2, 1, 3] -> Expected: False (n=2, base=[1, 2, 2])
    nums1 = [2, 1, 3]
    print(f"Result for {nums1}: {sol.isGood(nums1)}")

    # Example 2: nums = [1, 3, 3, 2] -> Expected: True (n=3, base=[1, 2, 3, 3])
    nums2 = [1, 3, 3, 2]
    print(f"Result for {nums2}: {sol.isGood(nums2)}")

    # Example 3: nums = [1, 1] -> Expected: True (n=1, base=[1, 1])
    nums3 = [1, 1]
    print(f"Result for {nums3}: {sol.isGood(nums3)}")

    # Example 4: nums = [3, 4, 4, 1, 2, 1] -> Expected: False
    nums4 = [3, 4, 4, 1, 2, 1]
    print(f"Result for {nums4}: {sol.isGood(nums4)}")
