import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[] nums = { 5, 2, 3, 1 };
    int res = minimumPairRemoval(nums);
    System.out.println(res);
  }

  public static int minimumPairRemoval(int[] nums) {
    int n = nums.length;
    int count = 0;

    while (!isSorted(nums, n)) {
      n = sumAndMerge(nums, n);
      count++;
    }

    return count;
  }

  public static boolean isSorted(int[] arr, int n) {
    for (int i = 0; i < n - 1; i++) {
      if (arr[i] > arr[i + 1]) {
        return false;
      }
    }
    return true;
  }

  public static int sumAndMerge(int[] arr, int n) {
    int min = Integer.MAX_VALUE;
    int pos = -1;

    for (int i = 0; i < n - 1; i++) {
      int sum = arr[i] + arr[i + 1];

      if (sum < min) {
        min = sum;
        pos = i;
      }
    }

    for (int i = 0; i < n - 1; i++) {
      if (i == pos) {
        arr[i] += arr[i + 1];
      }

      if (i > pos) {
        arr[i] = arr[i + 1];
      }
    }

    return n - 1;
  }
}
