import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[] arr = { 1, 2, 3, 1, 3, 5, 8, 1, 9 };

    int res = findMaxSumOfNonAdjElements(arr);
    System.out.println(res);
  }

  static int findMaxSumOfNonAdjElements(int[] arr) {
    int n = arr.length;
    if (n == 0) return 0;
    if (n == 1) return arr[0];

    int prev2 = arr[0];
    int prev1 = Math.max(arr[0], arr[1]);

    for (int i = 2; i < n; i++) {
      int curr = Math.max(prev1, arr[i] + prev2);
      prev2 = prev1;
      prev1 = curr;
    }
    return prev1;
  }
}
