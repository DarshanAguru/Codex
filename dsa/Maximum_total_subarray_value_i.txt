# Problem: https://leetcode.com/problems/maximum-total-subarray-value-i/
# Leetcode Problem No: 3689
# Relevance: Array, Greedy

from typing import List

class Solution:

    # Logic: Greedy Approach (O(N) Time, O(1) Space)
    # 1. We want to select exactly `k` subarrays to maximize the sum of their values, where the value of a subarray is max - min.
    # 2. Since subarrays are allowed to overlap and can be chosen multiple times, the optimal strategy is to find the global maximum and global minimum of the array.
    # 3. The subarray spanning from the index of the global minimum to the global maximum will have the maximum possible value, which is `max(nums) - min(nums)`.
    # 4. We can choose this exact subarray `k` times.
    # 5. Therefore, the maximum total value is simply `(max(nums) - min(nums)) * k`.
    #
    # Dry run for nums = [2, 3, 1, 5, 4], k = 3:
    # 1. max(nums) = 5, min(nums) = 1.
    # 2. res = (5 - 1) * 3 = 4 * 3 = 12.
    # 3. Return 12.

    def maxTotalValue(self, nums: List[int], k: int) -> int:
        maxe = max(nums)
        mine = min(nums)
        res = (maxe - mine) * k
        return res

if __name__ == "__main__":
    sol = Solution()

    # Example 1: nums = [2, 3, 1, 5, 4], k = 3 -> Expected: 12
    nums1 = [2, 3, 1, 5, 4]
    k1 = 3
    print(f"Result: {sol.maxTotalValue(nums1, k1)}")

    # Example 2: nums = [1, 1, 1], k = 5 -> Expected: 0
    nums2 = [1, 1, 1]
    k2 = 5
    print(f"Result: {sol.maxTotalValue(nums2, k2)}")
