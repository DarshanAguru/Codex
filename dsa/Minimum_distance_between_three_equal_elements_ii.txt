# Problem: https://leetcode.com/problems/minimum-distance-between-three-equal-elements-ii/
# Leetcode Problem No: 3741
# Relevance: Array, Hash Table, Two Pointers

from typing import List

class Solution:

    # Logic: Next Occurrence Mapping
    # The problem defines distance as |i-j| + |j-k| + |k-i|. For three indices i < j < k 
    # where nums[i] == nums[j] == nums[k], the distance simplifies to 2 * (k - i).
    # 1. We want to find the minimum (k - i) among all triplets of equal elements.
    # 2. To do this efficiently, we can precompute the 'next' index for each position.
    # 3. We use a dictionary 'occurances' to keep track of the first seen index of each value 
    #    while iterating backwards.
    # 4. `next_idxs[i]` stores the index of the next occurrence of `nums[i]` after `i`.
    # 5. After building `next_idxs`, for each index `a`, we find `b = next_idxs[a]` 
    #    and `c = next_idxs[b]`.
    # 6. If both `b` and `c` exist, `(a, b, c)` is a valid triple of equal elements.
    # 7. We calculate `c - a` and keep track of the minimum.
    # 8. The final result is `min_diff * 2`, or -1 if no such triple exists.
    #
    # Dry run for nums = [1, 2, 1, 3, 1, 2, 2]:
    # 1. next_idxs (mapping i to next occurrence):
    #    [0:2, 1:5, 2:4, 3:-1, 4:-1, 5:6, 6:-1]
    # 2. Triples:
    #    - a=0: b=2, c=4. dist = 4-0 = 4.
    #    - a=1: b=5, c=6. dist = 6-1 = 5.
    #    - a=2: b=4, c=-1. (Stop)
    #    - a=5: b=6, c=-1. (Stop)
    # 3. Min diff = 4. Result = 4 * 2 = 8.
    def minimumDistance(self, nums: List[int]) -> int:
        n = len(nums)
        occurances = {}
        next_idxs = [-1] * n
        res = n + 1
        
        # Build next occurrences map
        for i in range(n - 1, -1, -1):
            if nums[i] in occurances:
                next_idxs[i] = occurances[nums[i]]
            occurances[nums[i]] = i
        
        # Traverse to find triplets (a, b, c) where a < b < c and nums[a] == nums[b] == nums[c]
        for a in range(n):
            b = next_idxs[a]
            if b != -1:
                c = next_idxs[b]
                if c != -1:
                    res = min(res, c - a)
        
        return -1 if res == n + 1 else res * 2

if __name__ == "__main__":
    sol = Solution()
    
    # Example 1
    nums1 = [1, 2, 1, 3, 1, 2, 2]
    print(f"Example 1: {sol.minimumDistance(nums1)}") # Expected: 8
    
    # Example 2
    nums2 = [1, 1, 1, 1]
    print(f"Example 2: {sol.minimumDistance(nums2)}") # Expected: 4
    
    # Example 3
    nums3 = [1, 2, 3]
    print(f"Example 3: {sol.minimumDistance(nums3)}") # Expected: -1
