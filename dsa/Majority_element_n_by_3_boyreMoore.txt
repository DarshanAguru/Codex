import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[] arr = { 11, 33, 11, 33, 11, 33 };

    // Logic: Boyer-Moore Voting Algorithm (Extended)
    // We want to find all elements that appear more than N/3 times.
    // There can be AT MOST 2 such elements (since 3 * (N/3 + 1) > N).
    // We maintain two candidate variables (`el1`, `el2`) and their counters (`cnt1`, `cnt2`).
    // 1. If current element matches a candidate, increment its counter.
    // 2. If a counter is 0, assign the current element to that candidate (if it's not the other candidate).
    // 3. If current element doesn't match either candidate and both counters > 0, decrement BOTH counters.
    // After finding the optimal candidates, we MUST do a second pass to verify if they actually
    // appear more than N/3 times.
    //
    // Dry run for [11, 33, 11, 33, 11, 33]:
    // i=0(11): el1=11, c1=1.
    // i=1(33): el2=33, c2=1.
    // i=2(11): matches el1 -> c1=2.
    // i=3(33): matches el2 -> c2=2.
    // i=4(11): matches el1 -> c1=3.
    // i=5(33): matches el2 -> c2=3.
    // Verification pass: 11 appears 3 times (3 > 6/3=2). 33 appears 3 times (3 > 2).
    // Both added to result.
    int cnt1 = 0,
      cnt2 = 0;
    int el1 = Integer.MIN_VALUE;
    int el2 = Integer.MIN_VALUE;

    // 1. Find candidates
    for (int i = 0; i < arr.length; i++) {
      if (cnt1 == 0 && el2 != arr[i]) {
        cnt1 = 1;
        el1 = arr[i];
      } else if (cnt2 == 0 && el1 != arr[i]) {
        cnt2 = 1;
        el2 = arr[i];
      } else if (arr[i] == el1) {
        cnt1++;
      } else if (arr[i] == el2) {
        cnt2++;
      } else {
        cnt1--;
        cnt2--;
      }
    }

    // 2. Verify candidates
    List<Integer> ls = new ArrayList<>();
    cnt1 = 0;
    cnt2 = 0;

    for (int i = 0; i < arr.length; i++) {
      if (arr[i] == el1) cnt1++;
      if (arr[i] == el2) cnt2++;
    }

    int minimum = (int) (arr.length / 3) + 1;

    if (cnt1 >= minimum) ls.add(el1);
    if (cnt2 >= minimum) ls.add(el2);

    for (int i : ls) {
      System.out.print(i + " ");
    }
  }
}
