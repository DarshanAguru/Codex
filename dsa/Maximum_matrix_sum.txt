// Problem: https://leetcode.com/problems/maximum-matrix-sum/
// Leetcode Problem No: 1975
// Relevance: Matrix, Greedy, Array

import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[][] matrix = { { 1, 2, 3 }, { -1, -2, -3 }, { 1, 2, 3 } };
    long res = maxMatrixSum(matrix);
    System.out.println("Result: " + res);
  }

  // Logic: Matrix Parity & Greedy
  // We have an operation that can multiply any two adjacent elements by -1.
  // This operation effectively allows us to "move" a negative sign anywhere in the matrix!
  // It also allows us to cancel out pairs of negative signs (-1 * -1 = 1).
  // Therefore:
  // 1. If there is an EVEN number of negative numbers in the entire matrix, we can pair
  //    them all up and turn every single number positive. The max sum is the sum of all absolute values.
  // 2. If there is an ODD number of negative numbers, exactly ONE negative sign will remain.
  //    We can move this single negative sign to the element with the SMALLEST absolute value
  //    in the entire matrix to minimize its impact. The max sum is the sum of all absolute values,
  //    except we subtract the smallest absolute value twice (once to remove it from the sum,
  //    and once to subtract it as a negative).
  //
  // Dry run for matrix = [[1,2,3], [-1,-2,-3], [1,2,3]]:
  // Total absolute sum: 1+2+3 + 1+2+3 + 1+2+3 = 18.
  // Negatives count: 3 (odd).
  // Smallest absolute value: 1.
  // Since negatives is odd, subtract 2 * minVal: 18 - 2(1) = 16.
  // Returns 16.
  static long maxMatrixSum(int[][] matrix) {
    long sum = 0;
    int minVal = Integer.MAX_VALUE;
    int negatives = 0;

    for (int[] row : matrix) {
      for (int val : row) {
        sum += Math.abs(val);
        if (val < 0) {
          negatives++;
        }
        minVal = Math.min(minVal, Math.abs(val));
      }
    }

    // If odd negatives, we must subtract twice the smallest absolute value
    if (negatives % 2 != 0) {
      sum -= 2 * minVal;
    }

    return sum;
  }
}
