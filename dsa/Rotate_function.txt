# Problem: https://leetcode.com/problems/rotate-function/
# Leetcode Problem No: 396
# Relevance: Array, Math, Dynamic Programming

from typing import List

class Solution:

    # Logic: Mathematical relationship between F(i) and F(i-1)
    # Let S be the sum of all elements in nums.
    # F(0) = 0*nums[0] + 1*nums[1] + ... + (n-1)*nums[n-1]
    # F(1) = 0*nums[n-1] + 1*nums[0] + ... + (n-1)*nums[n-2]
    # F(1) - F(0) = (nums[0] + nums[1] + ... + nums[n-2]) - (n-1)*nums[n-1]
    # Adding and subtracting nums[n-1]:
    # F(1) - F(0) = (nums[0] + ... + nums[n-1]) - n*nums[n-1]
    # F(1) - F(0) = S - n*nums[n-1]
    # In general: F(i) = F(i-1) + S - n*nums[n-i]

    def maxRotateFunction(self, nums: List[int]) -> int:
        sumAll = sum(nums)
        n = len(nums)
        
        # Calculate F(0)
        current_f = sum(k * num for k, num in enumerate(nums))
        max_f = current_f
        
        # Iteratively calculate F(1) to F(n-1)
        for i in range(1, n):
            current_f = current_f + sumAll - n * nums[n - i]
            max_f = max(max_f, current_f)
            
        return max_f

if __name__ == "__main__":
    sol = Solution()

    # Example 1: nums = [4,3,2,6] -> Expected Result: 26
    nums1 = [4, 3, 2, 6]
    print(f"Result for [4, 3, 2, 6]: {sol.maxRotateFunction(nums1)}")

    # Example 2: nums = [100] -> Expected Result: 0
    nums2 = [100]
    print(f"Result for [100]: {sol.maxRotateFunction(nums2)}")
