import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[][] points = { { 1, 1 }, { 3, 4 }, { -1, 0 } };
    System.out.println(minTimeToVisitAllPoints(points));
  }

  // Logic: Chebyshev Distance / Math
  // We want to find the minimum time to travel between a series of 2D points.
  // We can move exactly 1 unit horizontally, vertically, or diagonally in 1 second.
  // Moving diagonally changes both X and Y by 1.
  // The time taken to travel between `Point A (x1, y1)` and `Point B (x2, y2)` is simply the
  // *maximum* of the absolute differences in their X and Y coordinates.
  // This is known as Chebyshev distance: `max(|x2 - x1|, |y2 - y1|)`.
  // We sum these distances for all consecutive points in the array.
  //
  // Dry run for points = [[1,1], [3,4], [-1,0]]:
  // 1. [1,1] to [3,4]: dx = |3-1|=2, dy = |4-1|=3. max(2,3) = 3 seconds.
  // 2. [3,4] to [-1,0]: dx = |-1-3|=4, dy = |0-4|=4. max(4,4) = 4 seconds.
  // Total time = 3 + 4 = 7.
  public static int minTimeToVisitAllPoints(int[][] points) {
    int res = 0;
    for (int i = 0; i < points.length - 1; i++) {
      res += Math.max(
        Math.abs(points[i + 1][1] - points[i][1]),
        Math.abs(points[i + 1][0] - points[i][0])
      );
    }
    return res;
  }
}
