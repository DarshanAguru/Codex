# Date: 05/07/2026
# Problem: https://leetcode.com/problems/number-of-paths-with-max-score/
# Leetcode Problem No: 1301
# Relevance: Array, Dynamic Programming, Matrix

from typing import List

class Solution:

    # Logic: Dynamic Programming (O(N^2) Time, O(N^2) Space)
    # 1. We want to find the maximum score we can collect from bottom-right (n-1, n-1) to top-left (0, 0),
    #    along with the number of paths that achieve this maximum score.
    # 2. We can move from (i, j) to (i-1, j), (i, j-1), or (i-1, j-1).
    #    Equivalently, working backward from bottom-right, we can transition to (i, j) from its neighbors:
    #    (i+1, j), (i, j+1), and (i+1, j+1) (which are bottom, right, and bottom-right neighbors).
    # 3. We maintain two 2D grids:
    #    - `dp[i][j]`: stores the maximum score to reach (i, j) from the end (n-1, n-1). Initialized to -1.
    #    - `ways[i][j]`: stores the number of paths to reach (i, j) with that maximum score. Initialized to 0.
    # 4. Base cases:
    #    - `dp[n-1][n-1] = 0`
    #    - `ways[n-1][n-1] = 1`
    # 5. We iterate backward from bottom-right to top-left. For each cell `(i, j)`:
    #    - If `board[i][j] == 'X'`, it's blocked, so we skip it.
    #    - Otherwise, we look at the valid neighbor cells `(ni, nj)` to see which one offers the maximum `dp` score.
    #    - If there's a unique maximum `dp[ni][nj]`, we set `best = dp[ni][nj]` and `cnt = ways[ni][nj]`.
    #    - If there are multiple neighbors with the same maximum score, we sum their `ways` values modulo `10^9 + 7`.
    #    - If `best != -1`, then `dp[i][j] = best + value(board[i][j])` (value is 0 for start 'E' / destination 'S', and `int(char)` for numeric cells)
    #      and `ways[i][j] = cnt`.
    # 6. Finally, if `dp[0][0] == -1`, it's impossible to reach (0, 0), so we return `[0, 0]`. Otherwise, we return `[dp[0][0], ways[0][0]]`.
    #
    # Dry run for board = ["E23", "2X2", "12S"]:
    # n = 3, MOD = 10^9 + 7
    # dp = [[-1, -1, -1], [-1, -1, -1], [-1, -1, 0]]
    # ways = [[0, 0, 0], [0, 0, 0], [0, 0, 1]]
    #
    # (i, j) transitions:
    # - (2, 1) ('2'):
    #   - Neighbors: (2, 2) has dp = 0, ways = 1.
    #   - best = 0, cnt = 1.
    #   - dp[2][1] = 0 + 2 = 2. ways[2][1] = 1.
    #
    # - (1, 2) ('2'):
    #   - Neighbors: (2, 2) has dp = 0, ways = 1.
    #   - best = 0, cnt = 1.
    #   - dp[1][2] = 0 + 2 = 2. ways[1][2] = 1.
    #
    # - (1, 1) ('X'): blocked, skip.
    #
    # - (2, 0) ('1'):
    #   - Neighbors: (2, 1) has dp = 2, ways = 1. (2, 2) is out of bounds, (1, 1) is -1.
    #   - best = 2, cnt = 1.
    #   - dp[2][0] = 2 + 1 = 3. ways[2][0] = 1.
    #
    # - (1, 0) ('2'):
    #   - Neighbors: (2, 0) has dp = 3, ways = 1; (2, 1) has dp = 2, ways = 1; (1, 1) is -1.
    #   - best = 3, cnt = 1.
    #   - dp[1][0] = 3 + 2 = 5. ways[1][0] = 1.
    #
    # - (0, 2) ('3'):
    #   - Neighbors: (1, 2) has dp = 2, ways = 1; (1, 1) is -1.
    #   - best = 2, cnt = 1.
    #   - dp[0][2] = 2 + 3 = 5. ways[0][2] = 1.
    #
    # - (0, 1) ('2'):
    #   - Neighbors: (1, 1) is -1; (1, 2) has dp = 2, ways = 1; (0, 2) has dp = 5, ways = 1.
    #   - best = 5, cnt = 1.
    #   - dp[0][1] = 5 + 2 = 7. ways[0][1] = 1.
    #
    # - (0, 0) ('E'):
    #   - Neighbors: (1, 0) has dp = 5, ways = 1; (0, 1) has dp = 7, ways = 1; (1, 1) is -1.
    #   - best = 7, cnt = 1.
    #   - dp[0][0] = 7 + 0 = 7. ways[0][0] = 1.
    #
    # Result: [7, 1].

    def pathsWithMaxScore(self, board: List[str]) -> List[int]:
        n = len(board)
        MOD = 10 ** 9 + 7
        dirs = [(1, 0), (0, 1), (1, 1)]

        dp = [[-1] * n for _ in range(n)]
        ways = [[0] * n for _ in range(n)]

        dp[-1][-1] = 0
        ways[-1][-1] = 1

        for i in range(n-1, -1, -1):
            for j in range(n-1, -1, -1):
                if board[i][j] == 'X' or (i == n-1 and j == n-1):
                    continue
                
                best = -1
                cnt = 0

                for di, dj in dirs:
                    ni, nj = i + di, j + dj
                    if 0 <= ni < n and 0 <= nj < n and dp[ni][nj] != -1:
                        if dp[ni][nj] > best:
                            best = dp[ni][nj]
                            cnt = ways[ni][nj]
                        elif dp[ni][nj] == best:
                            cnt = (cnt + ways[ni][nj]) % MOD
                if best == -1:
                    continue
                
                val = 0
                if i != 0 or j != 0:
                    val = int(board[i][j])
                
                dp[i][j] = best + val
                ways[i][j] = cnt
        
        if dp[0][0] == -1:
            return [0, 0]
        
        return [dp[0][0], ways[0][0]]

if __name__ == "__main__":
    sol = Solution()

    # Example 1: board = ["E23","2X2","12S"] -> Expected: [7, 1]
    board1 = ["E23", "2X2", "12S"]
    print(f"Result for example 1: {sol.pathsWithMaxScore(board1)}")

    # Example 2: board = ["E12","1X1","21S"] -> Expected: [4, 2]
    board2 = ["E12", "1X1", "21S"]
    print(f"Result for example 2: {sol.pathsWithMaxScore(board2)}")

    # Example 3: board = ["E11","XXX","11S"] -> Expected: [0, 0]
    board3 = ["E11", "XXX", "11S"]
    print(f"Result for example 3: {sol.pathsWithMaxScore(board3)}")
