import java.lang.*;
import java.util.*;

class Main {
  public static void main(String[] args) {
    int k = 2;
    int w = 0;
    int[] profits = { 1, 2, 3 };
    int[] capital = { 0, 1, 1 };
    int res = findMaximizedCapital(k, w, profits, capital);
    System.out.println(res);
  }

  public static int findMaximizedCapital(int k, int w, int[] profits, int[] capital) {
    int n = profits.length;
    List<int[]> projs = new ArrayList<>();
    for (int i = 0; i < n; i++) {
      projs.add(new int[] { capital[i], profits[i] });
    }
    projs.sort((a, b) -> Integer.compare(a[0], b[0]));
    PriorityQueue<Integer> pq = new PriorityQueue<>(Collections.reverseOrder());
    int i = 0;

    while (k > 0) {
      while (i < n && projs.get(i)[0] <= w) {
        pq.offer(projs.get(i)[1]);
        i++;
      }
      if (pq.isEmpty()) {
        break;
      }
      w += pq.poll();
      k--;
    }
    return w;
  }
}
