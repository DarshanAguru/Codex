import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[] arr = { 12, 6, 4, 15, 17, 10 };
    int k = 6;

    // Minimize the maximum difference between heights

    // Logic: Sorting + Greedy Partitioning
    // We sort the array first. The original max difference is `arr[N-1] - arr[0]`.
    // We want to find a split point index `i`, where everything to the LEFT of `i`
    // (the smaller elements) increases by `k`, and everything to the RIGHT of `i`
    // (the larger elements) decreases by `k`.
    // If we split at index `i`:
    // - The new MAX height could be `arr[N-1] - k` (if rightmost is still largest)
    //   OR `arr[i] + k` (if the element we just increased became the largest).
    //   So, `maxi = Math.max(arr[N-1] - k, arr[i] + k)`.
    // - The new MIN height could be `arr[0] + k` (if leftmost is still smallest)
    //   OR `arr[i+1] - k` (if the element we just decreased became the smallest).
    //   So, `mini = Math.min(arr[0] + k, arr[i+1] - k)`.
    // We iterate over all possible split points `i` and keep track of the minimum
    // difference `maxi - mini`. (Ignore cases where a tower height becomes negative).
    //
    // Dry run for arr = [12, 6, 4, 15, 17, 10], k = 6:
    // Sort: [4, 6, 10, 12, 15, 17].
    // orig_diff = 17 - 4 = 13.
    // Base max = 17 - 6 = 11. Base min = 4 + 6 = 10.
    // i=0(val 4): mini=min(10, 6-6=0)=0. maxi=max(11, 4+6=10)=11. diff=11-0=11.
    // i=1(val 6): mini=min(10, 10-6=4)=4. maxi=max(11, 6+6=12)=12. diff=12-4=8.
    // i=2(val 10): mini=min(10, 12-6=6)=6. maxi=max(11, 10+6=16)=16. diff=16-6=10.
    // i=3(val 12): mini=min(10, 15-6=9)=9. maxi=max(11, 12+6=18)=18. diff=18-9=9.
    // i=4(val 15): mini=min(10, 17-6=11)=10. maxi=max(11, 15+6=21)=21. diff=21-10=11.
    // Result min_diff = min(13, 11, 8, 10, 9, 11) = 8.
    Arrays.sort(arr);

    int res = arr[arr.length - 1] - arr[0];

    // We consider two partitions: smaller elements increase by k, larger decrease by k
    int max = arr[arr.length - 1] - k;
    int min = arr[0] + k;

    for (int i = 0; i < arr.length - 1; i++) {
      // New possible min and max after splitting at index i
      int mini = Math.min(min, arr[i + 1] - k);
      int maxi = Math.max(max, arr[i] + k);

      // Height cannot be negative
      if (arr[i] - k < 0) continue;

      res = Math.min(res, maxi - mini);
    }

    System.out.println(res);
  }
}
