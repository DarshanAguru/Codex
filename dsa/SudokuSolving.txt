// Problem: https://leetcode.com/problems/sudoku-solver/
// Leetcode Problem No: 37
// Relevance: Array, Backtracking, Matrix

import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    char[][] board = fillBoard();

    System.out.println("INPUT:");
    printBoard(board);

    if (solveSudoku(board)) {
      System.out.println("OUTPUT:");
      printBoard(board);
    } else {
      System.out.println("No solution exists");
    }
  }

  // Logic: Backtracking
  // We want to fill the 9x9 Sudoku board. We iterate through every cell.
  // 1. If we find an empty cell ('.'), we try placing numbers '1' through '9'.
  // 2. For each number, we validate if it violates Sudoku rules (using `validate`).
  //    - It must not exist in the same row.
  //    - It must not exist in the same column.
  //    - It must not exist in the 3x3 subgrid.
  // 3. If valid, we place the number and recursively try to solve the rest of the board.
  // 4. If the recursive call returns true (solved), we propagate true upwards.
  // 5. If false, we BACKTRACK by resetting the cell to '.' and try the next number.
  // 6. If no number '1'-'9' works, we return false (meaning the earlier choices were wrong).
  // Once we pass all empty cells, we return true (Puzzle Solved!).
  //
  // Dry run (Conceptual):
  // At (0, 3), grid is '.'. Try '1'. Validator checks row 0, col 3, and top-middle 3x3 grid.
  // '1' exists in row 0. Try '2'. Valid! Place '2', recurse to next empty cell.
  // If down the line we hit a dead end, recursion returns false, we come back to (0, 3),
  // revert to '.', and try '6', etc.
  static boolean solveSudoku(char[][] board) {
    for (int i = 0; i < 9; i++) {
      for (int j = 0; j < 9; j++) {
        if (board[i][j] == '.') {
          for (char c = '1'; c <= '9'; c++) {
            if (validate(board, i, j, c)) {
              board[i][j] = c;

              if (solveSudoku(board)) {
                return true;
              } else {
                board[i][j] = '.'; // Backtrack
              }
            }
          }
          return false;
        }
      }
    }
    return true;
  }

  static boolean validate(char[][] board, int r, int c, char x) {
    for (int i = 0; i < 9; i++) {
      // Check Row
      if (board[r][i] == x) return false;

      // Check Column
      if (board[i][c] == x) return false;

      // Check 3x3 Subgrid
      if (board[3 * (r / 3) + i / 3][3 * (c / 3) + (i % 3)] == x) return false;
    }
    return true;
  }

  static char[][] fillBoard() {
    return new char[][] {
      { '9', '5', '7', '.', '1', '3', '.', '8', '4' },
      { '4', '8', '3', '.', '5', '7', '1', '.', '6' },
      { '.', '1', '2', '.', '4', '9', '5', '3', '7' },
      { '1', '7', '.', '3', '.', '4', '9', '.', '2' },
      { '5', '.', '4', '9', '7', '.', '3', '6', '.' },
      { '3', '.', '9', '5', '.', '8', '7', '.', '1' },
      { '8', '4', '5', '7', '9', '.', '6', '1', '3' },
      { '.', '9', '1', '.', '3', '6', '.', '7', '5' },
      { '7', '.', '6', '1', '8', '5', '4', '.', '9' },
    };
  }

  static void printBoard(char[][] board) {
    for (char[] row : board) {
      System.out.println(Arrays.toString(row));
    }
  }
}
