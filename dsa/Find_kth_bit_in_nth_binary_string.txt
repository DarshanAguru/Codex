// Relevance: String, Recursion

import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int n = 4;
    int k = 11;
    char res = findKthBit(n, k);
    System.out.println(res);
  }

  static char findKthBit(int n, int k) {
    if (n == 1) {
      return '0';
    }

    int length = (1 << n) - 1;
    int mid = (length / 2) + 1;

    if (k == mid) {
      return '1';
    } else if (k < mid) {
      return findKthBit(n - 1, k);
    } else {
      int mirrorK = length - k + 1;
      char bit = findKthBit(n - 1, mirrorK);
      return bit == '0' ? '1' : '0';
    }
  }
}
