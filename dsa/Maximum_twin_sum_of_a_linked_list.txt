from typing import Optional

# Definition for singly-linked list.
class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next

class Solution:

    # Logic: Slow/Fast Pointers + In-Place Reversal (O(N) Time, O(1) Space)
    # 1. We want to sum the i-th node and the (n-1-i)-th node. These are "twins".
    # 2. To do this in O(1) auxiliary space, we split the list into two halves.
    # 3. We use the slow and fast pointer approach to find the middle of the linked list (starts at `slow`).
    # 4. We reverse the second half of the linked list starting from `slow`.
    # 5. We then traverse both halves simultaneously using two pointers (`left` and `right`),
    #    calculating the twin sum and updating the maximum twin sum seen so far (`res`).
    # 6. Finally, we return `res`.
    #
    # Dry run for head = [5, 4, 2, 1]:
    # 1. Finding middle:
    #    - Start: slow = 5, fast = 5.
    #    - Step 1: fast = 2, slow = 4.
    #    - Step 2: fast = None (loop ends), slow = 2.
    # 2. Reverse second half starting from slow (2 -> 1 -> None):
    #    - reverseList(slow) returns rev_head = 1 (1 -> 2 -> None).
    # 3. Traverse and sum:
    #    - left = 5, right = 1. Twin sum = 5 + 1 = 6. res = 6.
    #    - left = 4, right = 2. Twin sum = 4 + 2 = 6. res = max(6, 6) = 6.
    #    - left = None, right = None (loop ends).
    # 4. Return res = 6.

    def pairSum(self, head: Optional[ListNode]) -> int:
        slow = head
        fast = head
        while fast != None and fast.next != None:
            fast = fast.next.next
            slow = slow.next        
        
        def reverseList(h):
            temp = h
            prev = None
            curr = temp
            while temp != None:
                temp = temp.next
                curr.next = prev
                prev = curr
                curr = temp
            return prev
            
        rev_head = reverseList(slow)
        left = head
        right = rev_head
        res = 0
        while left != None and right != None:
            res = max(res, left.val + right.val)
            left = left.next
            right = right.next
        return res

if __name__ == "__main__":
    sol = Solution()

    # Helper to build list
    def build_list(arr):
        if not arr:
            return None
        head = ListNode(arr[0])
        curr = head
        for val in arr[1:]:
            curr.next = ListNode(val)
            curr = curr.next
        return head

    # Example 1: head = [5,4,2,1] -> Expected: 6
    head1 = build_list([5, 4, 2, 1])
    print(f"Result for [5, 4, 2, 1]: {sol.pairSum(head1)}")

    # Example 2: head = [4,2,2,3] -> Expected: 7
    head2 = build_list([4, 2, 2, 3])
    print(f"Result for [4, 2, 2, 3]: {sol.pairSum(head2)}")
