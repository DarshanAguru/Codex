import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[] digits = { 9, 9 };
    int[] res = plusOne(digits);
    System.out.println(Arrays.toString(res));
  }

  static int[] plusOne(int[] digits) {
    for (int i = digits.length - 1; i >= 0; i--) {
      if (digits[i] < 9) {
        digits[i]++;
        return digits;
      }
      digits[i] = 0;
    }

    int[] newNumber = new int[digits.length + 1];
    newNumber[0] = 1;
    return newNumber;
  }
}
