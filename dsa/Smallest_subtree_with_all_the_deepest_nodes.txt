// Date: 09/01/2026
// Problem: https://leetcode.com/problems/smallest-subtree-with-all-the-deepest-nodes/
// Leetcode Problem No: 865
// Relevance: Tree, Depth-First Search, Recursion

import java.lang.*;
import java.util.*;

class TreeNode {

  int val;
  TreeNode left;
  TreeNode right;

  TreeNode() {}

  TreeNode(int val) {
    this.val = val;
  }

  TreeNode(int val, TreeNode left, TreeNode right) {
    this.val = val;
    this.left = left;
    this.right = right;
  }
}

class Main {

  static class Pair {

    TreeNode node; // LCA of the deepest nodes in the subtree
    int val; // Function as the depth of the subtree

    Pair(TreeNode n, int v) {
      this.node = n;
      this.val = v;
    }
  }

  public static void main(String[] args) {
    // Example: [3,5,1,6,2,0,8,null,null,7,4]
    TreeNode root = new TreeNode(3);
    root.left = new TreeNode(5);
    root.right = new TreeNode(1);
    root.left.left = new TreeNode(6);
    root.left.right = new TreeNode(2);
    root.right.left = new TreeNode(0);
    root.right.right = new TreeNode(8);
    root.left.right.left = new TreeNode(7);
    root.left.right.right = new TreeNode(4);

    TreeNode result = subtreeWithAllDeepest(root);
    printTree(result);
  }

  public static void printTree(TreeNode root) {
    if (root == null) return;
    printTree(root.left);
    System.out.print(root.val + " ");
    printTree(root.right);
  }

  // Logic: DFS (Post-order Traversal)
  // We want the lowest common ancestor (LCA) of ALL the deepest nodes.
  // We use DFS to return a `Pair` containing:
  // 1. The deepest depth found in that subtree.
  // 2. The LCA node of the deepest nodes in that subtree.
  // At any given node:
  // - We recursively get the `(LCA, depth)` of the left and right subtrees.
  // - If `left.depth == right.depth`: This node balances the deepest nodes from both sides.
  //   So, THIS node is their LCA. Return `(this_node, left.depth + 1)`.
  // - If `left.depth > right.depth`: The deepest nodes are strictly in the left subtree.
  //   So, the LCA must also be in the left subtree. Return `(left.LCA, left.depth + 1)`.
  // - If `right.depth > left.depth`: Similar logic, return `(right.LCA, right.depth + 1)`.
  //
  // Dry run for root=3, L=5(depth=3), R=1(depth=2):
  // Let's look at node 2 (L=7, R=4). Both children return depth 1.
  //   Since 1 == 1, node 2 returns (Node(2), depth=2).
  // Now at node 5 (L=6, R=2): L returns (Node(6), 1), R returns (Node(2), 2).
  //   Since 2 > 1, right is deeper. Node 5 returns (Node(2), 3).
  // Now at root 3 (L=5, R=1): L returns (Node(2), 3), R returns (Node(0 or 8), 2).
  //   Since 3 > 2, left is deeper. Root 3 returns (Node(2), 4).
  // Final LCA is Node(2).
  public static TreeNode subtreeWithAllDeepest(TreeNode root) {
    Pair res = dfs(root);
    return res.node;
  }

  // Returns a pair containing the LCA of deepest nodes and the max depth in the current subtree
  private static Pair dfs(TreeNode node) {
    if (node == null) return new Pair(null, 0);

    Pair left = dfs(node.left);
    Pair right = dfs(node.right);

    // If left subtree is deeper, the target node is in the left subtree
    if (left.val > right.val) {
      return new Pair(left.node, left.val + 1);
    }
    // If right subtree is deeper, the target node is in the right subtree
    else if (right.val > left.val) {
      return new Pair(right.node, right.val + 1);
    }
    // If both subtrees are equally deep, current node is the LCA
    else {
      return new Pair(node, left.val + 1);
    }
  }
}
