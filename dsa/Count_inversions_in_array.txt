// Date: 06/08/2025
// Problem: https://leetcode.com/problems/count-the-number-of-inversions/
// Leetcode Problem No: 3193 (Note: Classic Inversion Count is often GFG; related LC #493)
// Relevance: Array, Merge Sort, Divide and Conquer

import java.lang.*;
import java.util.*;

class Main {

  public static void main(String[] args) {
    int[] arr = { 5, 4, 3, 2, 1 };
    int res = findInversions(arr);
    System.out.println(res);
  }

  // Logic: Merge Sort based Divide and Conquer
  // An inversion is a pair (i, j) where i < j and arr[i] > arr[j].
  // We can count inversions efficiently while doing a Merge Sort.
  // During the merge step, if `arr[low] <= arr[high]`, no inversion is formed.
  // BUT if `arr[low] > arr[high]`, it means `arr[high]` is smaller than all remaining elements
  // in the left half (from `low` to `mid` inclusive) since the left half is already sorted.
  // Thus, the number of inversions involving `arr[high]` is exactly `(mid - low + 1)`.
  //
  // Dry run for arr = [5, 4, 3, 2, 1]:
  // Ultimately merges will combine sorted left and right halves.
  // E.g. left=[4, 5], right=[1, 2, 3].
  // low=0 (val 4), high=2 (val 1). 4 > 1. Inversions += (1 - 0 + 1) = 2. (Pairs (4,1), (5,1)).
  // temp = [1]. high=3 (val 2). 4 > 2. Inversions += 2. (Pairs (4,2), (5,2)).
  // temp = [1, 2]. high=4 (val 3). 4 > 3. Inversions += 2. (Pairs (4,3), (5,3)).
  // temp = [1, 2, 3].
  // Now 4 <= no more right elements. Remaining left elements are added.
  // Total inversions will be 10.
  static int findInversions(int[] arr) {
    return mergeSort(arr, 0, arr.length - 1);
  }

  static int mergeSort(int[] arr, int l, int r) {
    int cnt = 0;
    if (l >= r) return cnt;

    int mid = (l + r) / 2;
    cnt += mergeSort(arr, l, mid);
    cnt += mergeSort(arr, mid + 1, r);
    cnt += merge(arr, l, mid, r);

    return cnt;
  }

  static int merge(int[] arr, int l, int m, int r) {
    List<Integer> temp = new ArrayList<>();
    int low = l;
    int high = m + 1;

    int cnt = 0;

    while (low <= m && high <= r) {
      if (arr[low] <= arr[high]) {
        temp.add(arr[low]);
        low++;
      } else {
        // If Arr[low] > Arr[high], then all elements from low to m are greater than Arr[high]
        temp.add(arr[high]);
        cnt += (m - low + 1);
        high++;
      }
    }

    while (low <= m) {
      temp.add(arr[low]);
      low++;
    }

    while (high <= r) {
      temp.add(arr[high]);
      high++;
    }

    for (int i = l; i <= r; i++) {
      arr[i] = temp.get(i - l);
    }

    return cnt;
  }
}
