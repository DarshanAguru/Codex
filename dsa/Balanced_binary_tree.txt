// Leetcode Problem No: 110
// Relevance: Tree, Depth-First Search, Binary Tree

import java.lang.*;
import java.util.*;

class TreeNode {

  int val;
  TreeNode left;
  TreeNode right;

  TreeNode() {}

  TreeNode(int val) {
    this.val = val;
  }

  TreeNode(int val, TreeNode left, TreeNode right) {
    this.val = val;
    this.left = left;
    this.right = right;
  }
}

class Main {

  public static void main(String[] args) {
    // Example: [3,9,20,null,null,15,7]
    TreeNode root = new TreeNode(3);
    root.left = new TreeNode(9);
    root.right = new TreeNode(20);
    root.right.left = new TreeNode(15);
    root.right.right = new TreeNode(7);

    System.out.println(isBalanced(root));
  }

  // Logic: DFS (Post-order Traversal)
  // A binary tree is balanced if the heights of the two subtrees of any node never differ by more than 1.
  // Instead of recalculating heights from top to bottom (O(N^2)), we can calculate height from bottom up.
  // The helper function returns the height of the tree. If at any point it finds a subtree is unbalanced,
  // it returns -1.
  // If the left subtree or right subtree returns -1, we propagate -1 up.
  // If `Math.abs(leftH - rightH) > 1`, return -1. Otherwise, return `max(leftH, rightH) + 1`.
  //
  // Dry run for [3, 9, 20, null, null, 15, 7]:
  // helper(9) -> returns 1
  // helper(15) -> returns 1
  // helper(7) -> returns 1
  // helper(20): left=1, right=1. diff=0 <= 1. returns max(1,1)+1 = 2
  // helper(3): left=1 (from 9), right=2 (from 20). diff=1 <= 1. returns max(1,2)+1 = 3
  // Result is 3 (which is != -1), so tree is balanced (returns true).
  public static boolean isBalanced(TreeNode root) {
    return helper(root) != -1;
  }

  private static int helper(TreeNode root) {
    if (root == null) {
      return 0;
    }
    int left = helper(root.left);
    if (left == -1) {
      return -1;
    }
    int right = helper(root.right);
    if (right == -1) {
      return -1;
    }

    if (Math.abs(left - right) > 1) {
      return -1;
    }
    return Math.max(left, right) + 1;
  }
}
