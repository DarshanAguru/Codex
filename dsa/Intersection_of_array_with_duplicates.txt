// Problem: https://leetcode.com/problems/intersection-of-two-arrays/
// Leetcode Problem No: 349
// Relevance: Hash Table, Array

import java.util.*;

class Main {

  public static void main(String[] args) {
    int[] a = { 1, 2, 3, 2, 1 };
    int[] b = { 3, 2, 2, 3, 3, 2 };

    ArrayList<Integer> res = new ArrayList<>();

    // Store first array in HashSet for quick lookup (Unique elements only)
    HashSet<Integer> hs = new HashSet<>();
    for (int i : a) {
      hs.add(i);
    }

    // Iterate second array and check for intersection
    for (int i : b) {
      // Logic: If found in Set, add to result and remove from Set to prevent duplicates in result.
      // This logic produces Unique Intersection (Problem 349).
      if (hs.contains(i)) {
        res.add(i);
        hs.remove(i);
      }
    }

    printRes(res);
  }

  private static void printRes(ArrayList<?> arr) {
    // wildcard for cleaner generic warning safety
    System.out.print("[");
    for (int i = 0; i < arr.size(); i++) {
      if (i == arr.size() - 1) {
        System.out.print(arr.get(i));
      } else {
        System.out.print(arr.get(i) + ", ");
      }
    }
    System.out.println("]");
  }
}
