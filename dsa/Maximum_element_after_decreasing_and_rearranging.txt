# Leetcode Problem No: 1846
# Relevance: Array, Greedy, Sorting

from typing import List

class Solution:

    # Logic: Greedy with Sorting (O(N log N) Time, O(1) Space)
    # 1. Since we can rearrange elements, we sort `arr` first to process them in non-decreasing order.
    # 2. To satisfy the first condition, the first element must be 1, so we set `arr[0] = 1`.
    # 3. For subsequent elements, the difference between consecutive elements must be at most 1.
    #    To maximize the final element, we greedily keep each element as large as possible:
    #    `arr[i] = min(arr[i], arr[i-1] + 1)`.
    # 4. The maximum possible value will be the last element of the array.
    #
    # Dry run for arr = [100, 1, 1000]:
    # 1. Sort: arr = [1, 100, 1000]
    # 2. Set first element: arr[0] = 1
    # 3. Loop:
    #    - i = 1: arr[1] = min(100, arr[0] + 1) = min(100, 2) = 2. arr becomes [1, 2, 1000]
    #    - i = 2: arr[2] = min(1000, arr[1] + 1) = min(1000, 3) = 3. arr becomes [1, 2, 3]
    # 4. Return arr[-1] = 3.

    def maximumElementAfterDecrementingAndRearranging(self, arr: List[int]) -> int:
        arr.sort()
        arr[0] = 1
        for i in range(1, len(arr)):
            arr[i] = min(arr[i], arr[i - 1] + 1)
        return arr[-1]

if __name__ == "__main__":
    sol = Solution()

    # Example 1: arr = [2, 2, 1, 2, 1] -> Expected: 2
    arr1 = [2, 2, 1, 2, 1]
    print(f"Result for {arr1}: {sol.maximumElementAfterDecrementingAndRearranging(arr1)}")

    # Example 2: arr = [100, 1, 1000] -> Expected: 3
    arr2 = [100, 1, 1000]
    print(f"Result for {arr2}: {sol.maximumElementAfterDecrementingAndRearranging(arr2)}")

    # Example 3: arr = [1, 2, 3, 4, 5] -> Expected: 5
    arr3 = [1, 2, 3, 4, 5]
    print(f"Result for {arr3}: {sol.maximumElementAfterDecrementingAndRearranging(arr3)}")
