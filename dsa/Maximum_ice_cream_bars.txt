# Problem: https://leetcode.com/problems/maximum-ice-cream-bars/
# Leetcode Problem No: 1833
# Relevance: Array, Greedy, Sorting, Counting Sort

from typing import List

class Solution:

    # Logic: Counting Sort / Frequency Array (O(N + M) Time, O(M) Space, where M is max(costs))
    # 1. We want to find the maximum number of ice cream bars we can buy with `coins` coins.
    # 2. To optimize without sorting the array, we can use a frequency array `freq` to count the occurrences of each cost.
    # 3. We find the maximum cost in `costs` to size the `freq` array.
    # 4. Populate `freq` by iterating through `costs`.
    # 5. Iterate through possible costs from 1 to `max(costs)`.
    #    - If `freq[cost]` is 0, skip.
    #    - Calculate the number of bars we can buy at the current cost: `min(freq[cost], coins // cost)`.
    #    - Add this count to our result `res`.
    #    - Deduct the corresponding cost from `coins`.
    # 6. Return `res`.
    #
    # Dry run for costs = [1, 3, 2, 4, 1], coins = 7:
    # max(costs) = 4, freq = [0, 2, 1, 1, 1], res = 0.
    # cost = 1: freq[1] = 2. min(2, 7 // 1) = 2. res = 2, coins = 7 - 2 * 1 = 5.
    # cost = 2: freq[2] = 1. min(1, 5 // 2) = 1. res = 3, coins = 5 - 1 * 2 = 3.
    # cost = 3: freq[3] = 1. min(1, 3 // 3) = 1. res = 4, coins = 3 - 1 * 3 = 0.
    # cost = 4: freq[4] = 1. min(1, 0 // 4) = 0. res = 4, coins = 0.
    # Returns 4.

    def maxIceCream(self, costs: List[int], coins: int) -> int:
        freq = [0] * (max(costs) + 1)

        for cost in costs:
            freq[cost] += 1

        res = 0

        for cost in range(1, max(costs) + 1):

            if freq[cost] == 0:
                continue

            res += min(freq[cost], coins // cost)

            coins -= min(freq[cost], coins // cost) * cost

        return res

if __name__ == "__main__":
    sol = Solution()

    # Example 1: costs = [1, 3, 2, 4, 1], coins = 7 -> Expected: 4
    costs1 = [1, 3, 2, 4, 1]
    coins1 = 7
    print(f"Result for {costs1} with coins={coins1}: {sol.maxIceCream(costs1, coins1)}")

    # Example 2: costs = [10, 6, 8, 7, 7, 8], coins = 5 -> Expected: 0
    costs2 = [10, 6, 8, 7, 7, 8]
    coins2 = 5
    print(f"Result for {costs2} with coins={coins2}: {sol.maxIceCream(costs2, coins2)}")

    # Example 3: costs = [1, 6, 3, 1, 2, 5], coins = 20 -> Expected: 6
    costs3 = [1, 6, 3, 1, 2, 5]
    coins3 = 20
    print(f"Result for {costs3} with coins={coins3}: {sol.maxIceCream(costs3, coins3)}")
