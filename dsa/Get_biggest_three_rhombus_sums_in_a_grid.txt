// Problem: https://leetcode.com/problems/get-biggest-three-rhombus-sums-in-a-grid/
// Leetcode Problem No: 1878
// Relevance: Array, Math, Matrix

import java.util.Arrays;
import java.util.TreeSet;

class Main {

  public static void main(String[] args) {
    int[][] grid = {
      { 3, 4, 5, 1, 3 },
      { 3, 3, 4, 2, 3 },
      { 20, 30, 200, 40, 10 },
      { 1, 5, 5, 4, 1 },
      { 4, 3, 2, 2, 5 },
    };
    System.out.println(Arrays.toString(getBiggestThree(grid)));
  }

  // Logic: Brute Force with TreeSet
  // The goal is to calculate the sums of all possible rhombuses in the grid and return the top 3 distinct largest.
  // 1. A TreeSet is used to automatically sift out duplicate sum values and maintain them in a sorted order.
  // 2. We iterate through every cell `(i, j)` in the grid, considering it the center of the rhombus.
  // 3. For each cell, we evaluate all possible sizes `k` (from `k = 0` up to bounds limitation).
  //    - If `k == 0`, the rhombus consists of just the cell itself, so we add `grid[i][j]` directly.
  //    - For `k >= 1`, we expand outwards and verify if the corners `(i-k, i+k, j-k, j+k)` remain within the grid.
  // 4. A helper function computes the perimeter sum by iterating through the border edges of the rhombus:
  //    - (Top to Right), (Right to Bottom), (Bottom to Left), (Left to Top).
  // 5. Finally, we poll the largest items from the end of the `TreeSet` (up to 3 distinct maximum values)
  //    and form the result array.
  //
  // Dry run idea:
  // If the grid has multiple overlapping shapes, starting from `(2, 2)` with `k=1`, the helper calculates
  // the sum of the 4 surrounding diagonal cells (`grid[1][2]`, `grid[2][3]`, `grid[3][2]`, `grid[2][1]`),
  // adding it to the sorted set. Returning the results is handled by calling `pollLast()` three times,
  // guaranteeing descending order and exclusively unique numbers.
  public static int[] getBiggestThree(int[][] grid) {
    int m = grid.length;
    int n = grid[0].length;

    TreeSet<Integer> set = new TreeSet<>();

    for (int i = 0; i < m; i++) {
      for (int j = 0; j < n; j++) {
        set.add(grid[i][j]);

        int k = 1;
        while (!(i + k >= m || i - k < 0 || j + k >= n || j - k < 0)) {
          int sum = helper(i, j, k, grid);
          set.add(sum);
          k++;
        }
      }
    }

    int size = Math.min(3, set.size());
    int[] res = new int[size];
    for (int i = 0; i < size; i++) {
      res[i] = set.pollLast();
    }
    return res;
  }

  private static int helper(int i, int j, int k, int[][] grid) {
    int sum = 0;

    int r = i - k;
    int c = j;
    for (int x = 0; x < k; x++) {
      sum += grid[r + x][c + x];
    }

    r = i;
    c = j + k;
    for (int x = 0; x < k; x++) {
      sum += grid[r + x][c - x];
    }

    r = i + k;
    c = j;
    for (int x = 0; x < k; x++) {
      sum += grid[r - x][c - x];
    }

    r = i;
    c = j - k;
    for (int x = 0; x < k; x++) {
      sum += grid[r - x][c + x];
    }

    return sum;
  }
}
