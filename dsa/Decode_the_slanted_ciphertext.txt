# Problem: https://leetcode.com/problems/decode-the-slanted-ciphertext/
# Leetcode Problem No: 2075
# Relevance: String, Simulation, Matrix

class Solution:

    # Logic: Diagonal Matrix Traversal
    # 1. The original text was written row by row and then read diagonally.
    # 2. To decode, we extract the characters diagonally by traversing row-by-row:
    #    - Start from each column 'j' of the first row (i=0).
    #    - Traverse diagonally down-right (i+1, k+1) while within boundaries.
    #    - The character at (i, k) is at index 'i * cols + k' in the encoded string.
    # 3. Join the characters and remove trailing spaces.
    #
    # Example for encodedText = "ch   ie   pr", rows = 3:
    # rows=3, n=12, cols=4
    # j=0: (0,0)='c' -> (1,1)='i' -> (2,2)='p' | res = ['c', 'i', 'p']
    # j=1: (0,1)='h' -> (1,2)='e' -> (2,3)='r' | res = ['c', 'i', 'p', 'h', 'e', 'r']
    # j=2: (0,2)=' ' -> (1,3)=' ' | res = ['c', 'i', 'p', 'h', 'e', 'r', ' ', ' ']
    # Result: "cipher"

    def decodeCiphertext(self, encodedText: str, rows: int) -> str:
        if rows == 1:
            return encodedText

        n = len(encodedText)
        cols = n // rows
        res = []

        # Read diagonally starting from each column of the first row
        for j in range(cols):
            i, k = 0, j
            while i < rows and k < cols:
                # Calculate the 1D index
                idx = i * cols + k
                if idx < n:
                    res.append(encodedText[idx])
                i += 1
                k += 1
        
        # rstrip() is crucial to remove trailing spaces used for matrix padding
        return "".join(res).rstrip()

if __name__ == "__main__":
    sol = Solution()
    
    # Example 1: encodedText = "ch   ie   pr", rows = 3
    encoded1 = "ch   ie   pr"
    rows1 = 3
    print(f"Example 1 Result: '{sol.decodeCiphertext(encoded1, rows1)}'")
    
    # Example 2: encodedText = "iveo    eed   l te   olc", rows = 4
    encoded2 = "iveo    eed   l te   olc"
    rows2 = 4
    print(f"Example 2 Result: '{sol.decodeCiphertext(encoded2, rows2)}'")
