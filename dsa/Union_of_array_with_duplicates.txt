import java.util.*;

class Main {
    public static void main(String[] args) {
        int[] a = {1, 2, 1, 1, 2};
        int[] b = {2, 2, 1, 2, 1};
        
        // Use HashSet to store unique elements from both arrays
        HashSet<Integer> hs = new HashSet<>();
        
        for (int i : a) {
            hs.add(i);
        }
        
        for (int i : b) {
            hs.add(i);
        }
        
        // Convert Set to Array
        int[] res = new int[hs.size()];
        int l = 0;
        for (int i : hs) {
            res[l++] = i;
        }
        
        printRes(res);
    }
    
    private static void printRes(int[] arr) {
        System.out.print("[");
        for (int i = 0; i < arr.length; i++) {
            if (i == arr.length - 1) {
                System.out.print(arr[i]);
            } else {
                System.out.print(arr[i] + ", ");
            }
        }
        System.out.println("]");
    }
}
