# Problem: XOR After Range Multiplication Queries I
# Leetcode Problem No: N/A (Custom/Contest)
# Relevance: Array, Simulation, XOR

from typing import List

class Solution:
    _MOD = 10**9 + 7

    # Logic: Direct Range Simulation
    # 1. Process each query (s, e, k, v) where:
    #    - s: Start index
    #    - e: End index
    #    - k: Step size
    #    - v: Multiplier
    # 2. For each query, multiply nums[i] by v for every k-th index in [s, e].
    # 3. Apply modulo operation at each multiplication step to prevent overflow.
    # 4. After all queries are processed, compute the cumulative XOR sum of the array.
    # 
    # Dry run for nums = [1, 2, 3], queries = [[0, 2, 1, 2]]:
    # 1. Query [0, 2, 1, 2]:
    #    - i=0: nums[0] = (1 * 2) % MOD = 2
    #    - i=1: nums[1] = (2 * 2) % MOD = 4
    #    - i=2: nums[2] = (3 * 2) % MOD = 6
    # 2. XOR Sum Calculation:
    #    - res = 0 ^ 2 = 2
    #    - res = 2 ^ 4 = 6
    #    - res = 6 ^ 6 = 0
    # 3. Return 0.

    def xorAfterQueries(self, nums: List[int], queries: List[List[int]]) -> int:
        for query in queries:
            s, e, k, v = query
            for i in range(s, e+1, k):
                nums[i] = (nums[i] * v) % self._MOD
        
        res = 0
        for num in nums:
            res ^= num

        return res

if __name__ == "__main__":
    sol = Solution()
    
    # Example 1
    nums1 = [1, 2, 3]
    queries1 = [[0, 2, 1, 2]]
    print(f"Example 1: {sol.xorAfterQueries(nums1, queries1)}") # Expected: 0
    
    # Example 2
    nums2 = [2, 4, 6]
    queries2 = [[0, 1, 1, 3], [1, 2, 1, 1]]
    # Q1: nums[0]=6, nums[1]=12 -> [6, 12, 6]
    # Q2: nums[1]=12, nums[2]=6 -> [6, 12, 6]
    # XOR: 6 ^ 12 ^ 6 = 12
    print(f"Example 2: {sol.xorAfterQueries(nums2, queries2)}") # Expected: 12
