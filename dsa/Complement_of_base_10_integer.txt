// Problem: Complement of Base 10 Integer
// Relevance: Bit Manipulation

class Main {

  public static void main(String[] args) {
    int n = 5;
    System.out.println(bitwiseComplement(n));
  }

  // Logic: Bit Manipulation
  // The complement of a base-10 integer is obtained by flipping all the bits
  // in its binary representation. This is equivalent to `XOR`ing the number with a mask
  // of all '1's of the SAME bit length as the number.
  // 1. We find the bit length (number of bits) of `n` using `Integer.toBinaryString(n).length()`.
  // 2. We create a mask of that length with all 1s: `(1 << bitCount) - 1`.
  //    (Example: if bit length is 3, mask is (1 << 3) - 1 = 8 - 1 = 7, which is `111` in binary).
  // 3. We return `mask ^ n`, which flips exactly the relevant bits of `n`.
  //
  // Dry run for n = 5 (binary `101`):
  // 1. bitCount of "101" is 3.
  // 2. mask = 1 << 3 = 8 (binary `1000`).
  // 3. mask = 8 - 1 = 7 (binary `111`).
  // 4. Return 7 ^ 5 (`111` ^ `101`) = 2 (`010`).
  public static int bitwiseComplement(int n) {
    int mask = 1;
    int bitCount = Integer.toBinaryString(n).length();
    mask = mask << bitCount;
    mask = mask - 1;
    return mask ^ n;
  }
}
